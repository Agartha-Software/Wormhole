14pub async fn handle_connection<Stream>(pods: &mut HashMap<String, Pod>, mut stream: Stream) -> bool
15where
16    Stream: tokio::io::AsyncWrite + tokio::io::AsyncRead + Unpin,
17{
18    log::debug!("Connection recieved");
19
20    let size = stream
21        .read_u32()
22        .await
23        .expect("Failed to read recieved command, shouldn't be possible!");
24    log::trace!("waiting for {size} bytes!");
25    let mut buffer: Vec<u8> = Vec::with_capacity(size as usize);
26    let _size = stream
27        .read_buf(&mut buffer)
28        .await
29        .expect("Failed to read recieved command, shouldn't be possible!");
30
31    log::trace!("found {_size} bytes!");
32
33    match bincode::deserialize::<Command>(&buffer) {
34        Ok(command) => handle_command(command, pods, stream)
35            .await
36            .unwrap_or_else(|e| {
37                log::error!("Network Error: {e:?}"); // TODO verify relevance
38                false
39            }),
40        Err(err) => {
41            log::error!("Command recieved not recognized by the service: {err:?}");
42            eprintln!("Command recieved not recognized by the service.");
43            false
44        }
45    }
46}

48pub async fn send_answer<T, Stream>(answer: T, stream: &mut Stream) -> std::io::Result<()>
49where
50    T: Serialize,
51    Stream: tokio::io::AsyncWrite + tokio::io::AsyncRead + Unpin,
52{
53    let serialized =
54        bincode::serialize(&answer).expect("Can't serialize cli answer, shouldn't be possible!");
55
56    stream.write_u32(serialized.len() as u32).await?;
57    stream.write_all(&serialized).await
58}

60async fn handle_command<Stream>(
61    command: Command,
62    pods: &mut HashMap<String, Pod>,
63    mut stream: Stream,
64) -> std::io::Result<bool>
65where
66    Stream: tokio::io::AsyncWrite + tokio::io::AsyncRead + Unpin,
67{
68    let stream = &mut stream;
69
70    match command {
71        Command::Unfreeze(data) => unfreeze(data, stream).await,
72        Command::Freeze(data) => freeze(data, stream).await,
73        Command::New(data) => new(data, pods, stream).await,
74        Command::GetHosts(data) => gethosts(data, pods, stream).await,
75        Command::Inspect(data) => inspect(data, pods, stream).await,
76        Command::Remove(data) => remove(data, pods, stream).await,
77        Command::Status => status(stream).await,
78        Command::Tree(data) => tree(data, pods, stream).await,
79    }
