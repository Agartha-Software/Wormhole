1use crate::{ipc::error::TCPListenerError, network::ip::IpP};
2use tokio::net::TcpListener;
3
4const MAX_TRY_PORTS: u16 = 10;
5const MAX_PORT: u16 = 65535;
6const DEFAULT_ADDRESS: &str = "127.0.0.1:8081";
7
8pub async fn new_free_tcp_listener() -> Result<(TcpListener, String), TCPListenerError> {
9    let mut ip: IpP = IpP::try_from(DEFAULT_ADDRESS).expect("Invalid ip provided");
10
11    let mut port_tries_count = 0;
12    loop {
13        match TcpListener::bind(&ip.to_string()).await {
14            Ok(listener) => break Ok((listener, ip.to_string())),
15            Err(err) => {
16                if ip.port >= MAX_PORT {
17                    break Err(TCPListenerError::AboveMainPort { max_port: MAX_PORT });
18                }
19                if port_tries_count > MAX_TRY_PORTS {
20                    break Err(TCPListenerError::AboveMaxTry {
21                        max_try_port: MAX_TRY_PORTS,
22                    });
23                }
24                log::warn!("Address {ip} not available due to {err}, switching...",);
25                ip.set_port(ip.port + 1);
26                port_tries_count += 1;
27            }
28        }
29    }
30}
