12pub async fn start_local_socket(socket: &str) -> io::Result<Stream> {
13    let name = if GenericNamespaced::is_supported() {
14        socket.to_ns_name::<GenericNamespaced>()?
15    } else {
16        //TODO: /tmp/ shouldn't be used, maybe use /var/run/ or /var/run/wormhole/
17        format!("/tmp/{socket}").to_fs_name::<GenericFilePath>()?
18    };
19    Stream::connect(name).await
20}

22pub async fn send_command(command: Command, stream: &mut Stream) -> Result<(), std::io::Error> {
23    let serialized =
24        bincode::serialize(&command).expect("Can't serialize cli command, shouldn't be possible .");
25    log::trace!("Sending cmd: {:?}, bytes {}", command, serialized.len());
26
27    stream.write_u32(serialized.len() as u32).await?;
28    stream.write_all(&serialized).await
29}

31pub async fn recieve_answer<T>(stream: &mut Stream) -> Result<T, std::io::Error>
32where
33    T: for<'a> Deserialize<'a> + std::fmt::Debug,
34{
35    let size = stream.read_u32().await?;
36    let mut recived_answer = Vec::with_capacity(size as usize);
37
38    stream.read_buf(&mut recived_answer).await?;
39
40    bincode::deserialize::<T>(&recived_answer)
41        .map_err(|err| {
42            log::error!("Service answer deserialisation failed: {err}");
43            io::Error::new(
44                io::ErrorKind::InvalidData,
45                "Command recieved isn't recognized",
46            )
47        })
48        .inspect(|answer| log::trace!("Recieved answer: {:?}", answer))
49}
