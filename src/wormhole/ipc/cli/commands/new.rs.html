1use std::io;
2
3use interprocess::local_socket::tokio::Stream;
4
5use crate::{
6    cli::NewArgs,
7    ipc::{
8        answers::NewAnswer,
9        cli::connection::{recieve_answer, send_command},
10        commands::{Command, NewRequest},
11    },
12};
13
14pub async fn new(args: NewArgs, mut stream: Stream) -> io::Result<()> {
15    let mountpoint = match args.mountpoint {
16        Some(mountpoint) => Ok(mountpoint),
17        None => std::env::current_dir().map(|path| path.join(args.name.clone())),
18    }?;
19
20    let NewArgs {
21        name,
22        port,
23        url,
24        hostname,
25        listen_url,
26        additional_hosts,
27        ..
28    } = args;
29
30    let request = NewRequest {
31        mountpoint,
32        name: name.clone(),
33        port,
34        url,
35        hostname,
36        listen_url,
37        additional_hosts,
38    };
39    send_command(Command::New(request), &mut stream).await?;
40
41    match recieve_answer::<NewAnswer>(&mut stream).await? {
42        NewAnswer::Success(new_port) => {
43            println!("Pod '{name}' created with success with port '{new_port}'.");
44            Ok(())
45        }
46        NewAnswer::AlreadyExist => Err(io::Error::new(
47            io::ErrorKind::AddrInUse,
48            "Pod already exist, couldn't create.",
49        )),
50        NewAnswer::InvalidIp => Err(io::Error::new(
51            io::ErrorKind::AddrInUse,
52            "Given port is already used.",
53        )),
54        NewAnswer::BindImpossible(e) => {
55            eprintln!("Failed to bind the given pod:");
56            Err(e.into())
57        }
58        NewAnswer::FailedToCreatePod(e) => {
59            eprintln!("Failed to create the given pod:");
60            Err(e.into())
61        }
62        NewAnswer::InvalidUrlIp => Err(io::Error::new(
63            io::ErrorKind::InvalidInput,
64            "Url given to connect to is invalid.",
65        )),
66    }
67}
