58fn inode_to_fuse_fileattr(inode: Inode) -> FileAttr {
59    let mut attr: FileAttr = inode.meta.into();
60    attr.ino = inode.id;
61    attr.kind = match inode.entry {
62        FsEntry::Directory(_) => fuser::FileType::Directory,
63        FsEntry::File(_) => fuser::FileType::RegularFile,
64    };
65    attr
66}

145    fn getxattr(
146        &mut self,
147        _req: &Request<'_>,
148        ino: u64,
149        name: &OsStr,
150        size: u32,
151        reply: ReplyXattr,
152    ) {
153        let attr = self
154            .fs_interface
155            .get_inode_xattr(ino, &name.to_string_lossy().to_string());
156
157        let data = match attr {
158            Ok(data) => data,
159            Err(GetXAttrError::KeyNotFound) => {
160                reply.error(libc::ERANGE);
161                return;
162            }
163            Err(GetXAttrError::WhError { source }) => {
164                reply.error(source.to_libc());
165                return;
166            }
167        };
168
169        if size == 0 {
170            reply.size(data.len() as u32);
171        } else {
172            reply.data(&data);
173        }
174    }

176    fn setxattr(
177        &mut self,
178        _req: &Request<'_>,
179        ino: u64,
180        name: &OsStr,
181        data: &[u8],
182        flags: i32,
183        _position: u32, // Postion undocumented
184        reply: ReplyEmpty,
185    ) {
186        // As we follow linux implementation in spirit, data size limit at 64kb
187        if data.len() > 64000 {
188            return reply.error(libc::ENOSPC);
189        }
190
191        let key = name.to_string_lossy().to_string();
192
193        if flags == XATTR_CREATE || flags == XATTR_REPLACE {
194            match self.fs_interface.xattr_exists(ino, &key) {
195                Ok(true) => {
196                    if flags == XATTR_CREATE {
197                        return reply.error(libc::EEXIST);
198                    }
199                }
200                Ok(false) => {
201                    if flags == XATTR_REPLACE {
202                        return reply.error(libc::ENODATA);
203                    }
204                }
205                Err(err) => {
206                    return reply.error(err.to_libc());
207                }
208            }
209        }
210
211        // TODO - Implement After permission implementation
212        // let attr = self.fs_interface.get_inode_attributes(ino);
213        // if attr.unwrap().perm == valid {
214        //     reply.error(libc::EPERM);
215        // }
216
217        match self
218            .fs_interface
219            .network_interface
220            .set_inode_xattr(ino, key, data.to_vec())
221        {
222            Ok(_) => reply.ok(),
223            Err(err) => reply.error(err.to_libc()),
224        }
225    }

244    fn listxattr(&mut self, _req: &Request<'_>, ino: u64, size: u32, reply: ReplyXattr) {
245        match self.fs_interface.list_inode_xattr(ino) {
246            Ok(keys) => {
247                let mut bytes = vec![];
248
249                for key in keys {
250                    bytes.extend(key.bytes());
251                    bytes.push(0);
252                }
253                if size == 0 {
254                    reply.size(bytes.len() as u32);
255                } else if size >= bytes.len() as u32 {
256                    reply.data(&bytes);
257                } else {
258                    reply.error(libc::ERANGE)
259                }
260                return;
261            }
262            Err(err) => match err.to_libc() {
263                libc::ENOENT => reply.error(libc::EBADF), // Not found became Bad file descriptor
264                or => reply.error(or),
265            },
266        }
267    }

269    fn read(
270        &mut self,
271        _req: &Request,
272        ino: u64,
273        file_handle: u64,
274        offset: i64,
275        size: u32,
276        _flags: i32,
277        _lock: Option<u64>,
278        reply: ReplyData,
279    ) {
280        let mut buf = vec![];
281        buf.resize(size as usize, 0);
282        match self.fs_interface.read_file(
283            ino,
284            offset.try_into().expect("read::read offset negative"),
285            &mut buf,
286            file_handle,

288            Ok(size) => {
289                buf.resize(size, 0);
290                reply.data(&buf)
291            }
292            Err(ReadError::WhError { source }) => reply.error(source.to_libc()),
293            Err(ReadError::PullError {
294                source: PullError::WhError { source },
295            }) => reply.error(source.to_libc()),
296            Err(ReadError::CantPull) => reply.error(libc::ENETUNREACH),
297            Err(ReadError::LocalReadFailed { io }) => reply.error(
298                io.raw_os_error()
299                    .expect("Local read error should always be the underling libc::open os error"),
300            ),
301            Err(ReadError::PullError {
302                source: PullError::NoHostAvailable,
303            }) => reply.error(libc::ENETUNREACH),
304            Err(ReadError::NoFileHandle) => reply.error(libc::EBADFD), // Shouldn't happend
305            //According to the man EBADF if the fd is not a valid file descriptor or is not open for reading.
306            Err(ReadError::NoReadPermission) => reply.error(libc::EBADFD),

310    fn readdir(
311        &mut self,
312        _req: &Request,
313        ino: u64,
314        _fh: u64,
315        offset: i64,
316        mut reply: ReplyDirectory,
317    ) {
318        let entries = match self.fs_interface.read_dir(ino) {
319            Ok(entries) => entries,
320            Err(e) => {
321                log::error!("readdir: ENOENT {e} {ino}");
322                reply.error(ENOENT);
323                return;
324            }
325        };
326
327        for (i, entry) in entries.into_iter().enumerate().skip(offset as usize) {
328            if reply.add(
329                ino,
330                // i + 1 means offset of the next entry
331                i as i64 + 1, // NOTE - in case of error, try i + 1
332                entry.entry.get_filetype().into(),
333                entry.name,
334            ) {
335                break;
336            }
337        }
338        reply.ok();
339    }

501            Err(OpenError::TruncReadOnly) => reply.error(libc::EACCES),

629    fn access(&mut self, _req: &Request<'_>, ino: u64, mask: i32, reply: ReplyEmpty) {
630        let meta = match self.fs_interface.n_get_inode_attributes(ino) {
631            Ok(meta) => meta,
632            Err(err) => {
633                reply.error(err.to_libc());
634                return;
635            }
636        };
637
638        match AccessMode::from_libc(mask)
639            .and_then(|access| check_permissions(OpenFlags::from_libc(mask), access, meta.perm))
640        {
641            Ok(_) => reply.ok(),
642            Err(OpenError::MultipleAccessFlags) => reply.error(libc::EINVAL),
643            Err(OpenError::TruncReadOnly) => reply.error(libc::EACCES),
644            Err(OpenError::WrongPermissions) => reply.error(libc::EPERM),
645            Err(OpenError::WhError { source }) => reply.error(source.to_libc()),
646        };
647    }
