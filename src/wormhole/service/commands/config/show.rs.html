1use parking_lot::RwLock;
2use serde::Serialize;

5use crate::{

11
12async fn locking_config_to_string<Conf>(config: &Arc<RwLock<Conf>>) -> Option<String>
13where
14    Conf: Serialize + Clone,
15{
16    config.try_read_for(LOCK_TIMEOUT).map(|config| {
17        toml::to_string(&config.clone()).expect("Serialization shouldn't be able to fail")
18    })
19}
20
21fn generate_local_config_file(pod: &Pod) -> String {
22    let config = pod.generate_local_config();
23    toml::to_string(&config).expect("Serialization shouldn't be able to fail")
24}
25
26impl Service {
27    pub async fn show<Stream>(
28        &self,
29        args: PodId,
30        config_type: ConfigType,
31        stream: &mut either::Either<&mut Stream, &mut String>,
32    ) -> std::io::Result<()>
33    where
34        Stream: tokio::io::AsyncWrite + tokio::io::AsyncRead + Unpin,
35    {
36        match find_pod(&args, &self.pods) {
37            Some((_, pod)) => match config_type {
38                ConfigType::Local => {
39                    send_answer(
40                        ShowConfigAnswer::SuccessLocal(generate_local_config_file(pod)),
41                        stream,
42                    )
43                    .await
44                }
45                ConfigType::Global => match locking_config_to_string(&pod.global_config).await {
46                    Some(global) => {
47                        send_answer(ShowConfigAnswer::SuccessGlobal(global), stream).await
48                    }
49                    None => send_answer(ShowConfigAnswer::ConfigBlock, stream).await,
50                },
51                ConfigType::Both => match locking_config_to_string(&pod.global_config).await {
52                    Some(global) => {
53                        send_answer(
54                            ShowConfigAnswer::SuccessBoth(generate_local_config_file(pod), global),
55                            stream,
56                        )
57                        .await
58                    }
59                    None => send_answer(ShowConfigAnswer::ConfigBlock, stream).await,
60                },
61            },
62            None => send_answer(ShowConfigAnswer::PodNotFound, stream).await,
63        }
64    }
65}
