1use std::sync::Arc;
2
3use crate::ipc::answers::RestartAnswer;
4use crate::ipc::commands::PodId;
5use crate::network::server::Server;
6use crate::pods::pod::Pod;
7use crate::service::commands::{find_frozen_pod, find_pod};
8use crate::service::connection::send_answer;
9use crate::service::Service;
10
11impl Service {
12    pub async fn restart<Stream>(
13        &mut self,
14        id: PodId,
15        stream: &mut either::Either<&mut Stream, &mut String>,
16    ) -> std::io::Result<()>
17    where
18        Stream: tokio::io::AsyncWrite + tokio::io::AsyncRead + Unpin,
19    {
20        let (name, pod) = match find_pod(&id, &self.pods) {
21            Some(found) => found,
22            None => {
23                return match find_frozen_pod(&id, &self.frozen_pods) {
24                    Some(_) => send_answer(RestartAnswer::PodFrozen, stream),
25                    None => send_answer(RestartAnswer::PodNotFound, stream),
26                }
27                .await;
28            }
29        };
30
31        let name = name.clone();
32
33        let proto = match pod.try_generate_prototype() {
34            Some(proto) => proto,
35            None => {
36                return send_answer(RestartAnswer::PodBlock, stream).await;
37            }
38        };
39
40        let pod = self
41            .pods
42            .remove(&name)
43            .expect("Already checked that the pod exist");
44
45        if let Err(err) = pod.stop().await {
46            return send_answer(RestartAnswer::PodStopFailed(err.into()), stream).await;
47        }
48
49        let server = match Server::from_specific_address(proto.bound_socket) {
50            Ok(server) => Arc::new(server),
51            Err(err) => {
52                return send_answer(RestartAnswer::CouldntBind(err.into()), stream).await;
53            }
54        };
55
56        match Pod::new(proto, server).await {
57            Ok(pod) => {
58                self.pods.insert(name.clone(), pod);
59                send_answer(RestartAnswer::Success(name), stream).await
60            }
61            Err(err) => send_answer(RestartAnswer::PodCreationFailed(err.into()), stream).await,
62        }
63    }
64}
