10fn name_from_string<'n>(name: &String) -> Result<Name<'n>, SocketListenerError> {
11    if GenericNamespaced::is_supported() {
12        name.clone().to_ns_name::<GenericNamespaced>()
13    } else {
14        format!("/tmp/{name}").to_fs_name::<GenericFilePath>()
15    }
16    .map_err(|io| SocketListenerError::InvalidAddr { io })
17}

19pub fn new_socket_listener(
20    specific_socket: Option<String>,
21) -> Result<(Listener, String), SocketListenerError> {
22    let name = specific_socket.unwrap_or(SOCKET_DEFAULT_NAME.to_string());
23    let ns_name = name_from_string(&name)?;
24    let listener = match ListenerOptions::new().name(ns_name).create_tokio() {
25        Err(e) if e.kind() == io::ErrorKind::AddrInUse => {
26            return Err(SocketListenerError::AddrInUse { name })
27        }
28        #[cfg(windows)]
29        Err(e) if e.kind() == io::ErrorKind::PermissionDenied => {
30            return Err(SocketListenerError::AddrInUse { name })
31        }
32        Err(e) => panic!("Unhandled socket error during listener creation: {e}"),
33        Ok(x) => x,
34    };
35
36    println!("Started Socket Listener at '{}'", name);
37    Ok((listener, name))
38}
