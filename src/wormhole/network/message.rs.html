15/// Message Content
16/// Represent the content of the intern message but is also the struct sent
17/// through the network
18#[derive(Serialize, Deserialize, Clone)]
19pub enum MessageContent {
20    Remove(InodeId),
21    Inode(Inode),
22    RequestFile(InodeId, Address),
23    PullAnswer(InodeId, Vec<u8>),
24    RedundancyFile(InodeId, Arc<Vec<u8>>),
25    /// Parent, New Parent, Name, New Name, overwrite
26    Rename(InodeId, InodeId, String, String, bool),
27    EditHosts(InodeId, Vec<Address>),
28    RevokeFile(InodeId, Address, Metadata),
29    AddHosts(InodeId, Vec<Address>),
30    RemoveHosts(InodeId, Vec<Address>),
31    EditMetadata(InodeId, Metadata),
32    SetXAttr(InodeId, String, Vec<u8>),
33    RemoveXAttr(InodeId, String),
34    RequestFs,
35    Disconnect(Address),
36
37    // (Arbo, peers, global_config)
38    FsAnswer(FileSystemSerialized, Vec<Address>, Vec<u8>),
39}
40
41impl fmt::Display for MessageContent {
42    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
43        let name = match self {
44            MessageContent::Remove(_) => "Remove",
45            MessageContent::Inode(_) => "Inode",
46            MessageContent::RequestFile(_, _) => "RequestFile",
47            MessageContent::PullAnswer(_, _) => "PullAnswer",
48            MessageContent::Rename(_, _, _, _, _) => "Rename",
49            MessageContent::EditHosts(_, _) => "EditHosts",
50            MessageContent::RevokeFile(_, _, _) => "RevokeFile",
51            MessageContent::AddHosts(_, _) => "AddHosts",
52            MessageContent::RemoveHosts(_, _) => "RemoveHosts",
53            MessageContent::EditMetadata(_, _) => "EditMetadata",
54            MessageContent::SetXAttr(_, _, _) => "SetXAttr",
55            MessageContent::RemoveXAttr(_, _) => "RemoveXAttr",
56            MessageContent::RequestFs => "RequestFs",
57            MessageContent::FsAnswer(_, _, _) => "FsAnswer",
58            MessageContent::RedundancyFile(_, _) => "RedundancyFile",
59            MessageContent::Disconnect(_) => "Disconnect",
60        };
61        write!(f, "{}", name)
62    }
63}

65impl fmt::Debug for MessageContent {
66    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
67        match self {
68            MessageContent::Inode(inode) => write!(
69                f,
70                "Inode({{{}, name: {}, parent:{}, {}}})",
71                inode.id,
72                inode.name,
73                inode.parent,
74                match inode.entry {
75                    crate::pods::arbo::FsEntry::File(_) => 'f',
76                    crate::pods::arbo::FsEntry::Directory(_) => 'd',
77                }
78            ),
79            MessageContent::RedundancyFile(id, _) => write!(f, "RedundancyFile({id}, <bin>)"),
80            MessageContent::FsAnswer(_, peers, _) => write!(f, "FsAnswer(<bin>, {peers:?}, <bin>"),
81            MessageContent::PullAnswer(id, _) => write!(f, "PullAnswer({id}, <bin>)"),
82            MessageContent::Remove(id) => write!(f, "Remove({id})"),
83            MessageContent::RequestFile(id, y) => write!(f, "RequestFile({id}, {y})"),
84            MessageContent::Rename(parent, new_parent, name, new_name, overwrite) => write!(
85                f,
86                "Rename(parent: {}, new_parent: {}, name: {}, new_name: {}, overwrite: {})",
87                parent, new_parent, name, new_name, overwrite
88            ),
89            MessageContent::EditHosts(id, hosts) => write!(f, "EditHosts({id}, {hosts:?})"),
90            MessageContent::RevokeFile(id, address, _) => {
91                write!(f, "RevokeFile({id}, {address}, <metadata>)")
92            }
93            MessageContent::AddHosts(id, hosts) => write!(f, "AddHosts({id}, {hosts:?})"),
94            MessageContent::RemoveHosts(id, hosts) => write!(f, "RemoveHosts({id}, {hosts:?})"),
95            MessageContent::EditMetadata(id, metadata) => {
96                write!(f, "EditMetadata({id}, {{ perm: {}}})", metadata.perm)
97            }
98            MessageContent::SetXAttr(id, name, data) => write!(
99                f,
100                "SetXAttr({id}, {name}, {}",
101                String::from_utf8(data.clone()).unwrap_or("<bin>".to_string())
102            ),
103            MessageContent::RemoveXAttr(id, name) => write!(f, "RemoveXAttr({id}, {name})"),
104            MessageContent::RequestFs => write!(f, "RequestFs"),
105            MessageContent::Disconnect(address) => write!(f, "Disconnect({address})"),
106        }
107    }
108}

129/// Message Going To Network
130/// Messages sent from fuser to process communicating to the peers
131#[derive(Debug)]
132pub enum ToNetworkMessage {
133    BroadcastMessage(MessageContent),
134    SpecificMessage(MessageAndStatus, Vec<Address>),
135}
136
137impl fmt::Display for ToNetworkMessage {
138    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
139        match self {
140            ToNetworkMessage::BroadcastMessage(content) => {
141                write!(f, "ToNetworkMessage::BroadcastMessage({})", content)
142            }
143            ToNetworkMessage::SpecificMessage((content, callback), adress) => {
144                write!(
145                    f,
146                    "ToNetworkMessage::SpecificMessage({}, callback: {}, to: {:?})",
147                    content,
148                    callback.is_some(),
149                    adress
150                )
151            }
152        }
153    }
154}
