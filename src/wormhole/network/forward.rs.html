13pub async fn forward_sender_to_peer<T>(
14    mut peer_write: T,
15    sender_out: &mut UnboundedReceiver<MessageAndStatus>,
16    peer: String,
17) where
18    T: Sink<Message> + Unpin,
19    <T as Sink<Message>>::Error: Debug,
20{
21    while let Some((message, status_tx)) = sender_out.recv().await {
22        let serialized = bincode::serialize::<MessageContent>(&message);
23        match serialized {
24            Err(e) => error!("failed to serialize toward {peer}: {e:?}"),
25            Ok(serialized) => {
26                let sent = peer_write.send(Message::binary(serialized)).await;
27
28                if let Err(e) = &sent {
29                    error!("failed to send to {peer}: {e:?}");
30                }
31
32                status_tx.inspect(|tx| {
33                    let _ = tx.send(sent.map_err(|_| WhError::NetworkDied {
34                        called_from: "forward_sender_to_write".to_string(),
35                    }));
36                });
37            }
38        }
39    }
40}

42pub async fn forward_peer_to_receiver<
43    T: StreamExt<Item = Result<Message, tokio_tungstenite::tungstenite::Error>>,
44>(
45    mut peer_read: SplitStream<T>,
46    receiver_in: UnboundedSender<FromNetworkMessage>,
47    peer: String,
48) {
49    while let Some(message) = peer_read.next().await {
50        match message {
51            Err(e) => error!("failed to read from {peer}: {e:?}"),
52            Ok(Message::Binary(message)) => {
53                let deserialized = bincode::deserialize::<MessageContent>(&message);
54                match deserialized {
55                    Err(e) => error!("failed to deserialize from {peer}: {e:?}"),
56                    Ok(deserialized) => {
57                        let sent = receiver_in.send(FromNetworkMessage {
58                            origin: peer.clone(),
59                            content: deserialized,
60                        });
61                        if let Err(e) = sent {
62                            error!("failed to forward message from {peer}: {e:?}");
63                        }
64                    }
65                }
66            }
67            other => warn!("unexpected message from {peer}: {other:?}"),
68        }
69    }
70
71    log::warn!("forward from {peer} closed by remote host");
72}
