16/// Ino is short for an Inode's id number. I can be thought as i-number (ino)
17/// This is the conventional name for inode's number across unix systems
18pub type Ino = u64;
19
20// root ino is always 1. Other ino are dynamically assigned
21pub const ROOT: Ino = 1;

23pub type XAttrs = HashMap<String, Vec<u8>>;
24
25#[derive(Debug, Serialize, Deserialize, Clone, PartialEq)]
26pub struct Inode {
27    pub parent: Ino,
28    pub id: Ino,
29    pub name: InodeName,
30    pub entry: FsEntry,
31    pub meta: Metadata,
32    pub xattrs: XAttrs,
33}
34
35impl Inode {
36    pub fn new(name: InodeName, parent_ino: Ino, id: Ino, entry: FsEntry, perm: u16) -> Self {
37        let meta = Metadata {
38            ino: id,
39            size: 0,
40            blocks: 0,
41            atime: SystemTime::now(),
42            mtime: SystemTime::now(),
43            ctime: SystemTime::now(),
44            crtime: SystemTime::now(),
45            kind: entry.get_filetype(),
46            perm,
47            nlink: 1 + matches!(entry, FsEntry::Directory(_)) as u32,
48            uid: 0,
49            gid: 0,
50            rdev: 0,
51            blksize: BLOCK_SIZE as u32,
52            flags: 0,
53        };
54
55        let xattrs = HashMap::new();
56
57        Self {
58            parent: parent_ino,
59            id,
60            name,
61            entry,
62            meta,
63            xattrs,
64        }
65    }
66}

73#[derive(Debug, Serialize, Deserialize, Clone, PartialEq, TS)]
74pub struct Metadata {
75    /// Inode number
76    pub ino: u64,
77    /// Size in bytes
78    pub size: u64,
79    /// Size in blocks
80    pub blocks: u64,
81    /// Time of last access
82    #[ts(as = "u64")]
83    pub atime: SystemTime,
84    /// Time of last modification
85    #[ts(as = "u64")]
86    pub mtime: SystemTime,
87    /// Time of last change
88    #[ts(as = "u64")]
89    pub ctime: SystemTime,
90    /// Time of creation (macOS only)
91    #[ts(as = "u64")]
92    pub crtime: SystemTime,
93    /// Kind of file (directory, file, pipe, etc)
94    pub kind: SimpleFileType,
95    /// Permissions
96    pub perm: u16,
97    /// Number of hard links
98    pub nlink: u32,
99    /// User id
100    pub uid: u32,
101    /// Group id
102    pub gid: u32,
103    /// Rdev
104    pub rdev: u32,
105    /// Block size
106    pub blksize: u32,
107    /// Flags (macOS only, see chflags(2))
108    pub flags: u32,
109}

111#[cfg(target_os = "linux")]
112impl TryInto<Metadata> for fs::Metadata {
113    type Error = std::io::Error;
114    fn try_into(self) -> Result<Metadata, std::io::Error> {
115        Ok(Metadata {
116            ino: 0, // TODO: unsafe default
117            size: self.len(),
118            blocks: 0,
119            atime: self.accessed()?,
120            mtime: self.modified()?,
121            ctime: self.modified()?,
122            crtime: self.created()?,
123            kind: if self.is_file() {
124                SimpleFileType::File
125            } else {
126                SimpleFileType::Directory
127            },
128            perm: self.permissions().mode() as u16,
129            nlink: self.nlink() as u32,
130            uid: self.uid(),
131            gid: self.gid(),
132            rdev: self.rdev() as u32,
133            blksize: self.blksize() as u32,
134            flags: 0,
135        })
136    }
137}
