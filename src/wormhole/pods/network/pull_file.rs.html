22impl NetworkInterface {
23    // REVIEW - recheck and simplify this if possible
24    pub fn pull_file_sync(&self, file: InodeId) -> Result<Option<Callback>, PullError> {
25        let self_addr = LocalConfig::read_lock(&self.local_config, "pull_file_sync")
26            .expect("pull_fyle_sync: can't get self_addr")
27            .general
28            .address
29            .clone();
30        let arbo = Arbo::n_read_lock(&self.arbo, "pull file sync")?;
31        let hosts = {
32            if let FsEntry::File(hosts) = &arbo.n_get_inode(file)?.entry {
33                hosts
34            } else {
35                panic!("Pulling a folder is invalid.")
36            }
37        };
38
39        assert!(hosts.len() != 0, "No hosts hold the file.");
40
41        if hosts.contains(&self_addr) {
42            // if the asked file is already on disk
43            Ok(None)
44        } else {
45            let callback = self.callbacks.n_create(Callback::Pull(file))?;
46            let (status_tx, mut status_rx) = tokio::sync::mpsc::unbounded_channel::<WhResult<()>>();
47
48            // will try to pull on all redundancies until success
49            for host in hosts {
50                // trying on host `pull_from`
51                self.to_network_message_tx
52                    .send(ToNetworkMessage::SpecificMessage(
53                        (
54                            MessageContent::RequestFile(file, self_addr.clone()),
55                            Some(status_tx.clone()),
56                        ),
57                        vec![host.clone()], // NOTE - naive choice for now
58                    ))
59                    .expect("pull_file: unable to request on the network thread");
60
61                // processing status
62                match status_rx
63                    .blocking_recv()
64                    .expect("pull_file: unable to get status from the network thread")
65                {
66                    Ok(()) => return Ok(Some(callback)),
67                    Err(_) => continue,
68                }
69            }
70            let _ = self.callbacks.resolve(callback, true);
71            log::error!("No host is currently able to send the file.\nFile: {file}");
72            return Err(PullError::NoHostAvailable);
73        }
74    }
