36fn into_boxed_io<T: std::error::Error>(err: T) -> io::Error {
37    std::io::Error::other(format!("{}: {err}", std::any::type_name::<T>()))

90    pub fn register_new_inode(&self, inode: Inode) -> Result<(), MakeInodeError> {
91        ITree::write_lock(&self.itree, "register_new_inode")?.add_inode(inode.clone())?;
92
93        if !ITree::is_local_only(inode.id) {
94            self.to_network_message_tx
95                .send(ToNetworkMessage::BroadcastMessage(MessageContent::Inode(
96                    inode,
97                )))
98                .expect("register inode: unable to update modification on the network thread");
99        }
100        Ok(())
101        // TODO - if unable to update for some reason, should be passed to the background worker
102    }

124    pub fn acknowledge_rename(
125        &self,
126        parent: Ino,
127        new_parent: Ino,
128        name: InodeName,
129        new_name: InodeName,
130    ) -> Result<(), RenameError> {
131        let mut itree = ITree::write_lock(&self.itree, "itree_rename_file")?;
132
133        itree
134            .mv_inode(parent, new_parent, name.as_ref(), new_name)
135            .map_err(|err| match err {
136                WhError::InodeNotFound => RenameError::DestinationParentNotFound,
137                WhError::InodeIsNotADirectory => RenameError::DestinationParentNotFolder,
138                source => RenameError::WhError { source },
139            })
140    }

142    /// Get a new inode, add the requested entry to the itree and inform the network
143    /// marks as reserved the Ino range up to the new Inode id
144    pub fn acknowledge_new_file(&self, inode: Inode) -> Result<(), MakeInodeError> {
145        let mut itree = ITree::write_lock(&self.itree, "acknowledge_new_file")?;
146        let _ = itree.mark_reserved_ino(inode.id); // this only happens in out-of-order handling of peer's inode creation, and isn't really an error
147        itree.add_inode(inode)
148    }

204    fn update_remote_hosts(&self, ino: Ino) -> WhResult<()> {
205        let inode = ITree::read_lock(&self.itree, "update_remote_hosts")?
206            .get_inode(ino)?
207            .clone();
208
209        if let FsEntry::File(hosts) = &inode.entry {
210            if !ITree::is_local_only(inode.id) {
211                self.to_network_message_tx
212                    .send(ToNetworkMessage::BroadcastMessage(
213                        MessageContent::EditHosts(inode.id, hosts.clone()),
214                    ))
215                    .expect(
216                        "update_remote_hosts: unable to update modification on the network thread",
217                    );
218            }
219            Ok(())
220        } else {
221            Err(WhError::InodeIsADirectory)
222        }
223    }

234    pub fn update_metadata(&self, id: Ino, meta: Metadata) -> WhResult<()> {
235        let mut itree = ITree::write_lock(&self.itree, "network_interface::update_metadata")?;
236        let mut fixed_meta = meta;
237        let ref_meta = &itree.get_inode(id)?.meta;
238
239        // meta's SystemTime is fragile: it can be silently corrupted such that
240        // serialization leads to a failure we can't deal with
241        if fixed_meta.atime.duration_since(UNIX_EPOCH).is_err() {
242            fixed_meta.atime = ref_meta.atime;
243        }
244
245        if fixed_meta.ctime.duration_since(UNIX_EPOCH).is_err() {
246            fixed_meta.ctime = ref_meta.ctime;
247        }
248
249        if fixed_meta.crtime.duration_since(UNIX_EPOCH).is_err() {
250            fixed_meta.crtime = ref_meta.crtime;
251        }
252
253        if fixed_meta.mtime.duration_since(UNIX_EPOCH).is_err() {
254            fixed_meta.mtime = ref_meta.mtime;
255        }
256
257        itree.set_inode_meta(id, fixed_meta.clone())?;
258        drop(itree);
259
260        if !ITree::is_local_only(id) {
261            self.to_network_message_tx
262                .send(ToNetworkMessage::BroadcastMessage(
263                    MessageContent::EditMetadata(id, fixed_meta),
264                ))
265                .expect("update_metadata: unable to update modification on the network thread");
266        }
267        Ok(())
268        /* REVIEW
269         * This system (and others broadcasts systems) should be reviewed as they don't check success.
270         * In this case, if another host misses this order, it will not update it's file.
271         * We could create a "broadcast" callback with the number of awaited confirmations and a timeout
272         * before resend or fail declaration.
273         * Or send a bunch of Specific messages
274         */
275    }

331    pub fn send_itree(&self, to: Address, global_config_bytes: Vec<u8>) -> WhResult<()> {
332        let clean_itree = ITree::read_lock(&self.itree, "send_itree")?
333            .clone()
334            .clean_local();
335        if let Some(peers) = self.peers.try_read_for(LOCK_TIMEOUT) {
336            let peers_address_list = peers
337                .iter()
338                .filter_map(|peer| {
339                    if peer.hostname != to {
340                        Some(peer.hostname.clone())
341                    } else {
342                        None
343                    }
344                })
345                .collect();
346
347            self.to_network_message_tx
348                .send(ToNetworkMessage::SpecificMessage(
349                    (
350                        MessageContent::FsAnswer(
351                            clean_itree,
352                            peers_address_list,
353                            global_config_bytes,
354                        ),
355                        None,
356                    ),
357                    vec![to],
358                ))
359                .expect("send_itree: unable to update modification on the network thread");
360            Ok(())
361        } else {
362            Err(WhError::WouldBlock {
363                called_from: "send_tree".to_owned(),
364            })
365        }
366    }

368    pub fn disconnect_peer(&self, addr: Address) -> WhResult<()> {
369        self.peers
370            .try_write_for(LOCK_TIMEOUT)
371            .ok_or(WhError::WouldBlock {
372                called_from: "disconnect_peer: can't write lock peers".to_owned(),
373            })?
374            .retain(|p| p.hostname != addr);
375
376        log::debug!("Disconnecting {addr}. Removing from inodes hosts");
377        for inode in ITree::write_lock(&self.itree, "disconnect_peer")?.inodes_mut() {
378            if let FsEntry::File(hosts) = &mut inode.entry {
379                hosts.retain(|h| *h != addr);
380            }
381        }
382        self.to_redundancy_tx
383            .send(RedundancyMessage::CheckIntegrity)
384            .unwrap();
385        Ok(())
386    }

388    pub async fn network_airport(
389        mut receiver: UnboundedReceiver<FromNetworkMessage>,
390        fs_interface: Arc<FsInterface>,
391    ) {
392        loop {
393            let FromNetworkMessage { origin, content } = match receiver.recv().await {
394                Some(message) => message,
395                None => continue,
396            };
397            if log::log_enabled!(log::Level::Debug) {
398                log::debug!("From {}: {:?}", origin, content);
399            } else {
400                log::info!("From {}: {}", origin, content);
401            }
402            let content_debug = format!("{content:?}");
403
404            let action_result = match content {
405                MessageContent::PullAnswer(id, binary) => fs_interface.recept_binary(id, binary)
406                                                            .map_err(into_boxed_io),
407                MessageContent::RedundancyFile(id, binary) => fs_interface.recept_redundancy(id, binary)
408                                            .map_err(into_boxed_io),
409                MessageContent::Inode(inode) => fs_interface.recept_inode(inode).map_err(into_boxed_io),
410                MessageContent::EditHosts(id, hosts) => fs_interface.recept_edit_hosts(id, hosts).map_err(into_boxed_io),
411                MessageContent::RevokeFile(id, host, meta) => fs_interface.recept_revoke_hosts(id, host, meta).map_err(into_boxed_io),
412                MessageContent::AddHosts(id, hosts) => fs_interface.recept_add_hosts(id, hosts).map_err(into_boxed_io),
413                MessageContent::RemoveHosts(id, hosts) => {
414                                            fs_interface.recept_remove_hosts(id, hosts).map_err(into_boxed_io)
415                                        }
416                MessageContent::EditMetadata(id, meta) =>
417                                            fs_interface.acknowledge_metadata(id, meta).map_err(into_boxed_io),
418                MessageContent::Remove(id) => fs_interface.recept_remove_inode(id).map_err(into_boxed_io),
419                MessageContent::RequestFile(inode) => fs_interface.send_file(inode, origin).map_err(into_boxed_io),
420                MessageContent::RequestFs => fs_interface.send_filesystem(origin).map_err(into_boxed_io),
421                MessageContent::Rename(parent, new_parent, name, new_name, overwrite) =>
422                                            fs_interface
423                                            .recept_rename(parent, new_parent, name, new_name, overwrite)
424                                            .map_err(into_boxed_io),
425                MessageContent::SetXAttr(ino, key, data) => fs_interface
426                                            .network_interface
427                                            .recept_inode_xattr(ino, &key, data)
428                                            .map_err(into_boxed_io),
429
430                MessageContent::RemoveXAttr(ino, key) => fs_interface
431                                            .network_interface
432                                            .recept_remove_inode_xattr(ino, &key)
433                                            .map_err(into_boxed_io),
434                MessageContent::FsAnswer(_, _, _) => {
435                                            Err(io::Error::new(ErrorKind::InvalidInput,
436                                                "Late answer from first connection, loaded network interface shouldn't recieve FsAnswer"))
437                                        },
438                MessageContent::Disconnect => fs_interface.network_interface.disconnect_peer(origin).map_err(into_boxed_io),
439                MessageContent::FileDelta(ino, meta, sig, delta) => fs_interface.accept_delta(ino, meta, sig, delta, origin)
440                                            .map_err(into_boxed_io),
441                MessageContent::FileChanged(ino, meta) => fs_interface.accept_file_changed(ino, meta, origin).map_err(into_boxed_io),
442                MessageContent::DeltaRequest(ino, sig) => fs_interface.respond_delta(ino, sig, origin).map_err(into_boxed_io),
443            };
444            if let Err(error) = action_result {
445                log::error!(
446                    "Network airport couldn't operate operation {content_debug}, error found: {error}"
447                );
448            }
449        }
450    }

452    pub async fn contact_peers(
453        peers_list: Arc<RwLock<Vec<PeerIPC>>>,
454        mut rx: UnboundedReceiver<ToNetworkMessage>,
455    ) {
456        log::info!("contact peers");
457        while let Some(message) = rx.recv().await {
458            // geeting all peers network senders
459            let peers_tx: Vec<(UnboundedSender<MessageAndStatus>, String)> = peers_list
460                .try_read_for(LOCK_TIMEOUT)
461                .expect("mutext error on contact_peers") // TODO - handle timeout
462                .iter()
463                .map(|peer| (peer.sender.clone(), peer.hostname.clone()))
464                .collect();
465
466            match message {
467                ToNetworkMessage::BroadcastMessage(message_content) => {
468                    peers_tx.iter().for_each(|(channel, address)| {
469                        channel
470                            .send((message_content.clone(), None))
471                            .unwrap_or_else(|e| {
472                                panic!("Failed to send message to peer {}: {e:?}", address)
473                            })
474                    });
475                }
476                ToNetworkMessage::SpecificMessage((message_content, status_tx), origins) => {
477                    let count = peers_tx
478                        .iter()
479                        .filter(|&(_, address)| origins.contains(address))
480                        .map(|(channel, address)| {
481                            channel
482                                .send((message_content.clone(), status_tx.clone())) // warning: only the first peer channel can set a status
483                                .unwrap_or_else(|e| {
484                                    panic!("Failed to send message to peer {}: {e:?}", address)
485                                })
486                        })
487                        .count();
488                    if count == 0 {
489                        log::warn!(
490                            "contact_peers: {message_content}: No peers by hostname {origins:?}"
491                        )
492                    }
493                }
494            };
495        }
496    }

498    pub async fn incoming_connections_watchdog(
499        server: Arc<Server>,
500        receiver_in: UnboundedSender<FromNetworkMessage>,
501        network_interface: Arc<NetworkInterface>,
502    ) {
503        while let Ok((stream, addr)) = server.listener.accept().await {
504            log::debug!("GOT ADDRESS {addr}");
505            let ws_stream = tokio_tungstenite::accept_async_with_config(
506                stream,
507                Some(
508                    WebSocketConfig::default()
509                        .max_message_size(None)
510                        .max_frame_size(None),
511                ),
512            )
513            .await
514            .expect("Error during the websocket handshake occurred");
515
516            match PeerIPC::accept(&network_interface, ws_stream, receiver_in.clone()).await {
517                Ok(new_peer) => network_interface
518                    .peers
519                    .try_write_for(LOCK_TIMEOUT)
520                    .expect("incoming_connections_watchdog: can't lock existing peers")
521                    .push(new_peer),
522                Err(e) => log::error!("incomming: accept: {e}"),
523            }
524        }
525    }
