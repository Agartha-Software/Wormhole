178        let inode_id = inode.id.clone();
179        Arbo::n_write_lock(&self.arbo, "register_new_inode")?.add_inode(inode.clone())?;
180
181        if !Arbo::is_local_only(inode_id) {
182            self.to_network_message_tx
183                .send(ToNetworkMessage::BroadcastMessage(MessageContent::Inode(

185                )))
186                .expect("register inode: unable to update modification on the network thread");
187        }
188        Ok(())
189        // TODO - if unable to update for some reason, should be passed to the background worker
190    }

276    fn affect_write_locally(&self, id: InodeId, new_size: usize) -> WhResult<Metadata> {
277        let mut arbo = Arbo::n_write_lock(&self.arbo, "network_interface.affect_write_locally")?;
278        let inode = arbo.n_get_inode_mut(id)?;
279        let new_size = (new_size as u64).max(inode.meta.size);
280        inode.meta.size = new_size as u64;
281        inode.meta.blocks = ((new_size + BLOCK_SIZE - 1) / BLOCK_SIZE) as u64;
282
283        inode.meta.mtime = SystemTime::now();
284
285        inode.entry = match &inode.entry {
286            FsEntry::File(_) => FsEntry::File(vec![self.hostname()?]),
287            _ => panic!("Can't edit hosts on folder"),
288        };
289        Ok(inode.meta.clone())
290    }

357    pub fn update_metadata(&self, id: InodeId, meta: Metadata) -> WhResult<()> {
358        let mut arbo = Arbo::n_write_lock(&self.arbo, "network_interface::update_metadata")?;
359        let mut fixed_meta = meta;
360        let ref_meta = &arbo.n_get_inode(id)?.meta;
361
362        // meta's SystemTime is fragile: it can be silently corrupted such that
363        // serialization leads to a failure we can't deal with
364        if fixed_meta.atime.duration_since(UNIX_EPOCH).is_err() {
365            fixed_meta.atime = ref_meta.atime;
366        }
367
368        if fixed_meta.ctime.duration_since(UNIX_EPOCH).is_err() {
369            fixed_meta.ctime = ref_meta.ctime;
370        }
371
372        if fixed_meta.crtime.duration_since(UNIX_EPOCH).is_err() {
373            fixed_meta.crtime = ref_meta.crtime;
374        }
375
376        if fixed_meta.mtime.duration_since(UNIX_EPOCH).is_err() {
377            fixed_meta.mtime = ref_meta.mtime;
378        }
379
380        arbo.n_set_inode_meta(id, fixed_meta.clone())?;
381        drop(arbo);
382
383        if !Arbo::is_local_only(id) {
384            self.to_network_message_tx
385                .send(ToNetworkMessage::BroadcastMessage(
386                    MessageContent::EditMetadata(id, fixed_meta),
387                ))
388                .expect("update_metadata: unable to update modification on the network thread");
389        }
390        Ok(())
391        /* REVIEW
392         * This system (and others broadcasts systems) should be reviewed as they don't check success.
393         * In this case, if another host misses this order, it will not update it's file.
394         * We could create a "broadcast" callback with the number of awaited confirmations and a timeout
395         * before resend or fail declaration.
396         * Or send a bunch of Specific messages
397         */
398    }

454    pub fn send_arbo(&self, to: Address, global_config_bytes: Vec<u8>) -> io::Result<()> {
455        let arbo = Arbo::read_lock(&self.arbo, "send_arbo")?;
456        let mut entries = arbo.get_raw_entries();
457
458        //Remove ignored entries
459        entries.retain(|ino, _| !Arbo::is_local_only(*ino));
460        entries.entry(1u64).and_modify(|inode| {
461            if let FsEntry::Directory(childrens) = &mut inode.entry {
462                childrens.retain(|x| !Arbo::is_local_only(*x));
463            }
464        });
465
466        if let Some(peers) = self.peers.try_read_for(LOCK_TIMEOUT) {
467            let peers_address_list = peers
468                .iter()
469                .filter_map(|peer| {
470                    if peer.hostname != to {
471                        Some(peer.hostname.clone())
472                    } else {
473                        None
474                    }
475                })
476                .collect();
477
478            self.to_network_message_tx
479                .send(ToNetworkMessage::SpecificMessage(
480                    (
481                        MessageContent::FsAnswer(
482                            FileSystemSerialized {
483                                fs_index: entries,
484                                next_inode: self.get_next_inode()?,
485                            },
486                            peers_address_list,
487                            global_config_bytes,
488                        ),
489                        None,
490                    ),
491                    vec![to],
492                ))
493                .expect("send_arbo: unable to update modification on the network thread");
494            Ok(())
495        } else {
496            Err(std::io::Error::new(
497                io::ErrorKind::WouldBlock,
498                "Deadlock while trying to read peers",
499            ))
500        }
501    }

503    pub fn disconnect_peer(&self, addr: Address) -> io::Result<()> {
504        self.peers
505            .try_write_for(LOCK_TIMEOUT)
506            .ok_or(std::io::Error::new(
507                std::io::ErrorKind::WouldBlock,
508                format!("disconnect_peer: can't write lock peers"),
509            ))?
510            .retain(|p| p.hostname != addr);
511
512        log::debug!("Disconnecting {addr}. Removing from inodes hosts");
513        for inode in Arbo::write_lock(&self.arbo, "disconnect_peer")?.inodes_mut() {
514            if let FsEntry::File(hosts) = &mut inode.entry {
515                hosts.retain(|h| *h != addr);
516            }
517        }
518        self.to_redundancy_tx
519            .send(RedundancyMessage::CheckIntegrity)
520            .unwrap();
521        Ok(())
522    }

524    pub async fn network_airport(
525        mut receiver: UnboundedReceiver<FromNetworkMessage>,
526        fs_interface: Arc<FsInterface>,
527    ) {
528        loop {
529            let FromNetworkMessage { origin, content } = match receiver.recv().await {
530                Some(message) => message,
531                None => continue,
532            };
533            if log::log_enabled!(log::Level::Debug) {
534                log::debug!("From {}: {:?}", origin, content);
535            } else {
536                log::info!("From {}: {}", origin, content);
537            }
538            let content_debug = format!("{content:?}");
539
540            let action_result = match content {
541                MessageContent::PullAnswer(id, binary) => fs_interface.recept_binary(id, binary),
542                MessageContent::RedundancyFile(id, binary) => fs_interface.recept_redundancy(id, binary)
543                    .map_err(|e| std::io::Error::new(
544                        std::io::ErrorKind::Other,
545                        format!("WhError: {e}"),
546                )),
547                MessageContent::Inode(inode) => fs_interface.recept_inode(inode).or_else(|err| {
548                        Err(std::io::Error::new(
549                            std::io::ErrorKind::Other,
550                            format!("WhError: {err}"),
551                        ))
552                    }),
553                MessageContent::EditHosts(id, hosts) => fs_interface.recept_edit_hosts(id, hosts).or_else(|err| {
554                        Err(std::io::Error::new(
555                            std::io::ErrorKind::Other,
556                            format!("WhError: {err}"),
557                        ))
558                    }),
559                MessageContent::RevokeFile(id, host, meta) => fs_interface.recept_revoke_hosts(id, host, meta).or_else(|err| {
560                        Err(std::io::Error::new(
561                            std::io::ErrorKind::Other,
562                            format!("WhError: {err}"),
563                        ))
564                    }),
565                MessageContent::AddHosts(id, hosts) => fs_interface.recept_add_hosts(id, hosts),
566                MessageContent::RemoveHosts(id, hosts) => {
567                    fs_interface.recept_remove_hosts(id, hosts)
568                }
569                MessageContent::EditMetadata(id, meta) =>
570                    fs_interface.acknowledge_metadata(id, meta).or_else(|err| {
571                        Err(std::io::Error::new(
572                            std::io::ErrorKind::Other,
573                            format!("WhError: {err}"),
574                        ))
575                    }),
576                MessageContent::Remove(id) => fs_interface.recept_remove_inode(id).or_else(|err| {
577                        Err(std::io::Error::new(
578                            std::io::ErrorKind::Other,
579                            format!("WhError: {err}"),
580                        ))
581                    }),
582                MessageContent::RequestFile(inode, peer) => fs_interface.send_file(inode, peer),
583                MessageContent::RequestFs => fs_interface.send_filesystem(origin),
584                MessageContent::Rename(parent, new_parent, name, new_name, overwrite) =>
585                    fs_interface
586                    .recept_rename(parent, new_parent, &name, &new_name, overwrite)
587                    .map_err(|err| {
588                        std::io::Error::new(
589                            std::io::ErrorKind::Other,
590                            format!("WhError: {err}"),
591                        )
592                    }),
593                MessageContent::SetXAttr(ino, key, data) => fs_interface
594                    .network_interface
595                    .recept_inode_xattr(ino, key, data)
596                    .or_else(|err| {
597                        Err(std::io::Error::new(
598                            std::io::ErrorKind::Other,
599                            format!("WhError: {err}"),
600                        ))
601                    }),
602                MessageContent::RemoveXAttr(ino, key) => fs_interface
603                    .network_interface
604                    .recept_remove_inode_xattr(ino, key)
605                    .or_else(|err| {
606                        Err(std::io::Error::new(
607                            std::io::ErrorKind::Other,
608                            format!("WhError: {err}"),
609                        ))
610                    }),
611                MessageContent::FsAnswer(_, _, _) => {
612                    Err(io::Error::new(ErrorKind::InvalidInput,
613                        "Late answer from first connection, loaded network interface shouldn't recieve FsAnswer"))
614                },
615                MessageContent::Disconnect(addr) => fs_interface.network_interface.disconnect_peer(addr)
616            };
617            if let Err(error) = action_result {
618                log::error!(
619                    "Network airport couldn't operate operation {content_debug}, error found: {error}"
620                );
621            }
622        }
623    }

625    pub async fn contact_peers(
626        peers_list: Arc<RwLock<Vec<PeerIPC>>>,
627        mut rx: UnboundedReceiver<ToNetworkMessage>,
628    ) {
629        log::info!("contact peers");
630        while let Some(message) = rx.recv().await {
631            // geeting all peers network senders
632            let peers_tx: Vec<(UnboundedSender<MessageAndStatus>, String)> = peers_list
633                .try_read_for(LOCK_TIMEOUT)
634                .expect("mutext error on contact_peers") // TODO - handle timeout
635                .iter()
636                .map(|peer| (peer.sender.clone(), peer.hostname.clone()))
637                .collect();
638
639            match message {
640                ToNetworkMessage::BroadcastMessage(message_content) => {
641                    peers_tx.iter().for_each(|(channel, address)| {
642                        channel
643                            .send((message_content.clone(), None))
644                            .expect(&format!("failed to send message to peer {}", address))
645                    });
646                }
647                ToNetworkMessage::SpecificMessage((message_content, status_tx), origins) => {
648                    peers_tx
649                        .iter()
650                        .filter(|&(_, address)| origins.contains(address))
651                        .for_each(|(channel, address)| {
652                            channel
653                                .send((message_content.clone(), status_tx.clone()))
654                                .expect(&format!("failed to send message to peer {}", address))
655                        });
656                }
657            };
658        }
659    }

661    pub async fn incoming_connections_watchdog(
662        server: Arc<Server>,
663        receiver_in: UnboundedSender<FromNetworkMessage>,
664        network_interface: Arc<NetworkInterface>,
665    ) {
666        while let Ok((stream, addr)) = server.listener.accept().await {
667            log::debug!("GOT ADDRESS {addr}");
668            let ws_stream = tokio_tungstenite::accept_async_with_config(
669                stream,
670                Some(
671                    WebSocketConfig::default()
672                        .max_message_size(None)
673                        .max_frame_size(None),
674                ),
675            )
676            .await
677            .expect("Error during the websocket handshake occurred");
678
679            match PeerIPC::accept(&network_interface, ws_stream, receiver_in.clone()).await {
680                Ok(new_peer) => network_interface
681                    .peers
682                    .try_write_for(LOCK_TIMEOUT)
683                    .expect("incoming_connections_watchdog: can't lock existing peers")
684                    .push(new_peer),
685                Err(e) => log::error!("incomming: accept: {e}"),
686            }
687        }
688    }
