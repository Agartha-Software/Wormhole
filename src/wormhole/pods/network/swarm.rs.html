1use crate::pods::network::behaviour::Behaviour;
2use libp2p::{
3    identify, noise,
4    request_response::{self, ProtocolSupport},
5    yamux, StreamProtocol, Swarm,
6};
7use std::{error::Error, time::Duration};
8
9const PROTOCOL_VERSION: &str = "/wormhole/1.0.0";
10
11pub const MAX_CONCURRENT_STREAMS: usize = 128;
12
13pub async fn create_swarm(nickname: String) -> Result<Swarm<Behaviour>, Box<dyn Error>> {
14    let swarm = libp2p::SwarmBuilder::with_new_identity()
15        .with_tokio()
16        .with_websocket(noise::Config::new, yamux::Config::default)
17        .await?
18        .with_behaviour(|key| {
19            let cfg = identify::Config::new(PROTOCOL_VERSION.to_string(), key.public())
20                .with_agent_version(nickname);
21
22            Behaviour {
23                request_response: request_response::Behaviour::new(
24                    [(StreamProtocol::new(PROTOCOL_VERSION), ProtocolSupport::Full)],
25                    request_response::Config::default()
26                        .with_max_concurrent_streams(MAX_CONCURRENT_STREAMS),
27                ),
28                identify: identify::Behaviour::new(cfg),
29            }
30        })?
31        .with_swarm_config(|c| c.with_idle_connection_timeout(Duration::from_secs(u64::MAX)))
32        .build();
33
34    Ok(swarm)
35}
