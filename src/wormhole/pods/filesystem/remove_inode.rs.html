11custom_error! {
12    /// Error describing the removal of a [Inode] from the [Arbo]
13    pub RemoveInodeError
14    WhError{source: WhError} = "{source}",
15    NonEmpty = "Can't remove non-empty dir",
16}

18custom_error! {
19    /// Error describing the removal of a [Inode] from the [Arbo] and the local file or folder
20    pub RemoveFileError
21    WhError{source: WhError} = "{source}",
22    NonEmpty = "Can't remove non-empty dir",
23    LocalDeletionFailed{io: std::io::Error} = "Local Deletion failed: {io}"
24}
25
26impl From<RemoveInodeError> for RemoveFileError {
27    fn from(value: RemoveInodeError) -> Self {
28        match value {
29            RemoveInodeError::WhError { source } => RemoveFileError::WhError { source },
30            RemoveInodeError::NonEmpty => RemoveFileError::NonEmpty,
31        }
32    }
33}
34
35impl FsInterface {
36    // NOTE - system specific (fuse/winfsp) code that need access to arbo or other classes
37    pub fn fuse_remove_inode(
38        &self,
39        parent: InodeId,
40        name: &std::ffi::OsStr,
41    ) -> Result<(), RemoveFileError> {
42        let target = {
43            let arbo = Arbo::n_read_lock(&self.arbo, "fs_interface::fuse_remove_inode")?;
44            let parent = arbo.n_get_inode(parent)?;
45            arbo.n_get_inode_child_by_name(parent, &name.to_string_lossy().to_string())?
46                .id
47        };
48
49        self.remove_inode(target)
50    }
51
52    pub fn remove_inode_locally(&self, id: InodeId) -> Result<(), RemoveFileError> {
53        let arbo = Arbo::n_read_lock(&self.arbo, "fs_interface::remove_inode")?;
54        let to_remove_path = arbo.n_get_path_from_inode_id(id)?;
55
56        match &arbo.n_get_inode(id)?.entry {
57            FsEntry::File(hosts) if hosts.contains(&self.network_interface.hostname()?) => self
58                .disk
59                .remove_file(&to_remove_path)
60                .map_err(|io| RemoveFileError::LocalDeletionFailed { io })?,
61            FsEntry::File(_) => {
62                // TODO: Remove when wormhole initialisation is cleaner
63                // try to delete the file even if it's not owned to prevent from conflicts on creation later
64                let _ = self.disk.remove_file(&to_remove_path);
65            }
66            FsEntry::Directory(children) if children.is_empty() => self
67                .disk
68                .remove_dir(&to_remove_path)
69                .map_err(|io| RemoveFileError::LocalDeletionFailed { io })?,
70            FsEntry::Directory(_) => return Err(RemoveFileError::NonEmpty),
71        };
72        Ok(())
73    }
74
75    pub fn remove_inode(&self, id: InodeId) -> Result<(), RemoveFileError> {
76        self.remove_inode_locally(id)?;
77        self.network_interface.unregister_inode(id)?;
78        Ok(())
79    }
80
81    pub fn recept_remove_inode(&self, id: InodeId) -> Result<(), RemoveFileError> {
82        self.remove_inode_locally(id)?;
83        self.network_interface.acknowledge_unregister_inode(id)?;
84        Ok(())
85    }
