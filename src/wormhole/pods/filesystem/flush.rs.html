31    /// Flush file changes to all connected peers
32    /// With a handle, the changed will be first attempted with the handle's base signature
33    /// the handle's signature will be flushed. The dirty flag will be reset
34    /// Without a handle, all peers will get a 'FileChanged' message and tracking peers will request the delta with their own base signature
35    /// Non-tracking peers will always only get a 'FileChanged' message
36    ///
37    pub fn flush(&self, ino: Ino, handle: Option<&mut FileHandle>) -> Result<(), FlushError> {
38        let peers = self.network_interface.peers.read();
39        let inode = self.network_interface.itree.read().get_inode(ino)?.clone();
40        let tracking = match &inode.entry {
41            FsEntry::File(tracking) => tracking,
42            _ => return Err(WhError::InodeIsADirectory.into()),
43        };
44        if let Some((signature, dirty)) =
45            handle.and_then(|h| h.signature.as_mut().map(|s| (s, &mut h.dirty)))
46        {
47            if !*dirty {
48                return Ok(());
49            }
50            let file = self.get_whole_file_sync(ino)?;
51            let delta = signature.diff(&file)?;
52
53            let old_sig = signature.clone();
54            *signature = Signature::new(&file)?;
55            *dirty = false;
56
57            for peer in peers.iter() {
58                if tracking.contains(peer) {
59                    self.network_interface
60                        .to_network_message_tx
61                        .send(ToNetworkMessage::SpecificMessage(
62                            Request::FileDelta(
63                                ino,
64                                inode.meta.clone(),
65                                old_sig.clone(),
66                                delta.clone(),
67                            ),
68                            vec![*peer],
69                        ))
70                        .map_err(|e| WhError::WouldBlock {
71                            called_from: e.to_string(),
72                        })?;
73                } else {
74                    self.network_interface
75                        .to_network_message_tx
76                        .send(ToNetworkMessage::SpecificMessage(
77                            Request::FileChanged(ino, inode.meta.clone()),
78                            vec![*peer],
79                        ))
80                        .map_err(|e| WhError::WouldBlock {
81                            called_from: e.to_string(),
82                        })?;
83                }
84            }
85        } else {
86            self.network_interface
87                .to_network_message_tx
88                .send(ToNetworkMessage::BroadcastMessage(Request::FileChanged(
89                    ino,
90                    inode.meta.clone(),
91                )))
92                .map_err(|e| WhError::WouldBlock {
93                    called_from: e.to_string(),
94                })?;
95        }
96        Ok(())
97    }

99    /// Apply a delta received from the network
100    /// deltas are in reference to a base signature, in case of signature mismatch
101    /// [Request::DeltaRequest] is emitted back to get the correct diff
102    ///
103    pub fn accept_delta(
104        &self,
105        ino: Ino,
106        meta: Metadata,
107        sig: Signature,
108        delta: Delta,
109    ) -> Result<Response, FlushError> {
110        log::trace!("accept_delta({ino})");
111        let file = match self.get_local_file(ino)? {
112            Some(file) => file,
113            None => {
114                log::warn!("accept_delta: received delta but isn't currently tracking the file!");
115                return Ok(Response::Success);
116            }
117        };
118        let local_sig = Signature::new_using(&file, sig.implementor())?;
119        log::trace!(
120            "signing <<\n{}\n>> = {:?}",
121            file.0.escape_ascii(),
122            local_sig
123        );
124        if local_sig == sig {
125            let patched = delta.patch(&file)?;
126            log::trace!(
127                "accept_delta: patched = {}",
128                String::from_utf8_lossy(&patched.0)
129            );
130
131            let itree = ITree::read_lock(&self.network_interface.itree, "fs_interface.write")?;
132            let path = itree.get_path_from_inode_id(ino)?;
133            drop(itree);
134
135            self.disk
136                .write_file(&path, &patched.0, 0)
137                .map_err(WriteError::from)?;
138            self.acknowledge_metadata(ino, meta).map_err(|e| match e {
139                AcknoledgeSetAttrError::WhError { source } => FlushError::from(source),
140                AcknoledgeSetAttrError::SetFileSizeIoError { io } => WriteError::from(io).into(),
141            })?;
142        } else {
143            log::warn!("accept_delta: signature does not match local sig!");
144            return Ok(Response::DeltaRequest(ino, local_sig));
145        }
146        Ok(Response::Success)
147    }

149    /// Acknowledge a file change and request the change contents if we are tracking the file
150    pub fn accept_file_changed(&self, ino: Ino, meta: Metadata) -> Result<Response, FlushError> {
151        self.acknowledge_metadata(ino, meta).map_err(|e| match e {
152            AcknoledgeSetAttrError::WhError { source } => FlushError::from(source),
153            AcknoledgeSetAttrError::SetFileSizeIoError { io } => WriteError::from(io).into(),
154        })?;
155        let file = match self.get_local_file(ino)? {
156            Some(file) => file,
157            None => return Ok(Response::Success),
158        };
159        let local_sig = Signature::new(&file)?;
160        Ok(Response::DeltaRequest(ino, local_sig))
161    }

163    pub fn respond_delta(
164        &self,
165        ino: Ino,
166        sig: Signature,
167        origin: PeerId,
168    ) -> Result<(), FlushError> {
169        let file = self.get_local_file(ino)?.ok_or(WhError::InodeNotFound)?;
170        let delta = sig.diff(&file)?;
171
172        let inode = self.network_interface.itree.read().get_inode(ino)?.clone();
173
174        self.network_interface
175            .to_network_message_tx
176            .send(ToNetworkMessage::SpecificMessage(
177                Request::FileDelta(ino, inode.meta, sig, delta),
178                vec![origin],
179            ))
180            .map_err(|e| WhError::WouldBlock {
181                called_from: e.to_string(),
182            })?;
183        Ok(())
184    }
