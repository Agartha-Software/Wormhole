22const FMODE_EXEC: i32 = 0x20;
23
24impl AccessMode {
25    #[cfg(target_os = "linux")]
26    pub fn from_libc(flags: i32) -> Result<AccessMode, OpenError> {
27        match flags & libc::O_ACCMODE {
28            libc::O_RDONLY => {
29                if flags & FMODE_EXEC != 0 {
30                    Ok(AccessMode::Execute)
31                } else {
32                    Ok(AccessMode::Read)
33                }
34            }
35            libc::O_WRONLY => Ok(AccessMode::Write),
36            libc::O_RDWR => Ok(AccessMode::ReadWrite),
37            //Exactly one access mode flag must be specified
38            _ => Err(OpenError::MultipleAccessFlags),
39        }
40    }
41}
42
43pub fn check_permissions(
44    flags: OpenFlags,
45    access: AccessMode,
46    inode_perm: u16,
47) -> Result<AccessMode, OpenError> {
48    match access {
49        AccessMode::Void => Ok(AccessMode::Void),
50        AccessMode::Read => {
51            if !has_read_perm(inode_perm) {
52                Err(OpenError::WrongPermissions)
53            //Behavior is undefined, but most filesystems return EACCES
54            } else if flags.trunc {
55                //EACCESS
56                Err(OpenError::TruncReadOnly)
57            //Open is from internal exec syscall
58            } else if flags.exec {
59                if !has_execute_perm(inode_perm) {
60                    Err(OpenError::WrongPermissions)
61                } else {
62                    Ok(AccessMode::Execute)
63                }
64            } else {
65                Ok(AccessMode::Read)
66            }
67        }
68        AccessMode::Write if !has_write_perm(inode_perm) => Err(OpenError::WrongPermissions),
69        AccessMode::Write => Ok(AccessMode::Write),
70        AccessMode::ReadWrite if !has_read_perm(inode_perm) || !has_write_perm(inode_perm) => {
71            Err(OpenError::WrongPermissions)
72        }
73        AccessMode::ReadWrite => Ok(AccessMode::ReadWrite),
74        AccessMode::Execute => {
75            if has_read_perm(inode_perm) {
76                Err(OpenError::WrongPermissions)
77            //Behavior is undefined, but most filesystems return EACCES
78            } else {
79                if !has_execute_perm(inode_perm) {
80                    Err(OpenError::WrongPermissions)
81                } else {
82                    Ok(AccessMode::Execute)
83                }
84            }
85        }
86    }
87}
