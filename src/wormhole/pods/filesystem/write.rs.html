13custom_error! {
14    /// Error describing the write syscall
15    pub WriteError
16    WhError{source: WhError} = "{source}",
17    LocalWriteFailed{io: std::io::Error} = "Local write failed: {io}",
18    NoFileHandle = "The file doesn't have a file handle",
19    NoWritePermission = "The permissions doesn't allow to write",
20}
21
22fn check_file_handle<'a>(
23    file_handles: &'a RwLockReadGuard<FileHandleManager>,
24    file_handle_id: UUID,
25) -> Result<&'a FileHandle, WriteError> {
26    match file_handles.handles.get(&file_handle_id) {
27        Some(&FileHandle {
28            perm: AccessMode::Read,
29            direct: _,
30            no_atime: _,
31        }) => return Err(WriteError::NoWritePermission),
32        Some(&FileHandle {
33            perm: AccessMode::Execute,
34            direct: _,
35            no_atime: _,
36        }) => return Err(WriteError::NoWritePermission),
37        None => return Err(WriteError::NoFileHandle),
38        Some(file_handle) => Ok(file_handle),
39    }
40}
41
42impl FsInterface {
43    pub fn write(
44        &self,
45        id: InodeId,
46        data: &[u8],
47        offset: usize,
48        file_handle: UUID,
49    ) -> Result<usize, WriteError> {
50        let file_handles = FileHandleManager::read_lock(&self.file_handles, "write")?;
51        let _file_handle = check_file_handle(&file_handles, file_handle)?;
52
53        let arbo = Arbo::n_read_lock(&self.arbo, "fs_interface.write")?;
54        let path = arbo.n_get_path_from_inode_id(id)?;
55        drop(arbo);
56
57        let new_size = offset + data.len();
58        let written = self
59            .disk
60            .write_file(&path, data, offset)
61            .map_err(|io| WriteError::LocalWriteFailed { io })?;
62
63        self.network_interface.write_file(id, new_size)?;
64        Ok(written)
65    }
