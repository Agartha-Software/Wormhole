34    pub fn acknowledge_metadata(
35        &self,
36        ino: Ino,
37        meta: Metadata,
38    ) -> Result<Response, AcknoledgeSetAttrError> {
39        let mut itree = ITree::write_lock(&self.network_interface.itree, "acknowledge_metadata")?;
40        let path = itree.get_path_from_inode_id(ino)?;
41        let inode = itree.get_inode_mut(ino)?;
42
43        if meta != inode.meta {
44            match &inode.entry {
45                FsEntry::Directory(_) if meta.size != inode.meta.size => {
46                    return Err(AcknoledgeSetAttrError::WhError {
47                        source: WhError::InodeIsADirectory,
48                    });
49                }
50                FsEntry::File(hosts) => {
51                    if hosts.contains(&self.network_interface.id) {
52                        let created = match &inode.entry {
53                            FsEntry::File(old_hosts) => {
54                                !old_hosts.contains(&self.network_interface.id)
55                            }
56                            _ => false,
57                        };
58                        if created {
59                            self.disk.new_file(&path, meta.perm).or_else(|io| {
60                                if io.kind() == std::io::ErrorKind::AlreadyExists {
61                                    Ok(())
62                                } else {
63                                    Err(AcknoledgeSetAttrError::SetFileSizeIoError { io })
64                                }
65                            })?;
66                        }
67                        if meta.size != inode.meta.size {
68                            self.disk
69                                .set_file_size(&path, meta.size as usize)
70                                .map_err(|io| AcknoledgeSetAttrError::SetFileSizeIoError { io })?;
71                        }
72                        if meta.perm != inode.meta.perm {
73                            self.disk
74                                .set_permisions(&path, meta.perm)
75                                .map_err(|io| AcknoledgeSetAttrError::SetFileSizeIoError { io })?;
76                        }
77                    }
78                }
79                _ => {
80                    if meta.perm != inode.meta.perm {
81                        self.disk
82                            .set_permisions(&path, meta.perm)
83                            .map_err(|io| AcknoledgeSetAttrError::SetFileSizeIoError { io })?;
84                    }
85                }
86            }
87        }
88
89        itree.get_inode_mut(ino)?.meta = meta;
90        Ok(Response::Success)
91    }

94    pub fn setattr(
95        &self,
96        ino: Ino,
97        mode: Option<u32>,
98        uid: Option<u32>,
99        gid: Option<u32>,
100        size: Option<u64>,
101        atime: Option<std::time::SystemTime>,
102        mtime: Option<std::time::SystemTime>,
103        ctime: Option<std::time::SystemTime>,
104        file_handle: Option<UUID>,
105        flags: Option<u32>,
106    ) -> Result<Metadata, SetAttrError> {
107        let itree = ITree::read_lock(&self.network_interface.itree, "setattr")?;
108        let path = itree.get_path_from_inode_id(ino)?;
109        let mut meta = itree.get_inode(ino)?.meta.clone();
110        drop(itree);
111
112        //Except for size, No permissions are required on the file itself, but permission is required on all of the directories in pathname that lead to the file.
113
114        // Extract specific informations from the file handle if it's defined
115        let (fh_perm, no_atime) = if let Some(file_handle) = file_handle {
116            let file_handler = FileHandleManager::read_lock(&self.file_handles, "setattr")?;
117            match file_handler.handles.get(&file_handle) {
118                None => return Err(SetAttrError::InvalidFileHandle),
119                Some(file_handle) => (Some(file_handle.perm), file_handle.no_atime),
120            }
121        } else {
122            (None, false)
123        };
124
125        if let Some(mode) = mode {
126            self.disk
127                .set_permisions(&path, mode as u16)
128                .map_err(|io| SetAttrError::SetPermIoError { io })?;
129            meta.perm = mode as u16;
130        }
131        // Set size if size it's defined, take permission from the file handle if the
132        if let Some(size) = size {
133            if meta.kind != SimpleFileType::File {
134                return Err(WhError::InodeIsADirectory.into());
135            }
136            match fh_perm {
137                Some(perm) if perm != AccessMode::Write && perm != AccessMode::ReadWrite => {
138                    return Err(SetAttrError::SizeNoPerm)
139                }
140                None if !has_write_perm(meta.perm) => return Err(SetAttrError::SizeNoPerm),
141                _ => {
142                    // In theory if size > meta.size, the file doesn't change in the memory but in case of read, the read should zero fill the rest of the file
143                    // But for now we don't support sparse file
144                    self.disk
145                        .set_file_size(&path, size as usize)
146                        .map_err(|io| SetAttrError::SetFileSizeIoError { io })?;
147                    meta.size = size;
148                    meta.blocks = size.div_ceil(BLOCK_SIZE);
149                }
150            };
151        }
152
153        if !no_atime {
154            if let Some(atime) = atime {
155                meta.atime = atime;
156            } else {
157                meta.atime = SystemTime::now();
158            }
159        }
160
161        if let Some(mtime) = mtime {
162            meta.mtime = mtime;
163        // Only size change represent a modification
164        } else if size.is_some() {
165            meta.mtime = SystemTime::now();
166        }
167
168        meta.ctime = ctime.unwrap_or(SystemTime::now());
169
170        //crtime is ignored because crtime is macos only and shouldn't be updated after file creation anyway
171        //
172        // REVIEW- we could implement this code for a perfect macos 1 to 1, but I think allowing such a feature
173        // on only one os is a very weird behavior
174        //
175        // if cfg!(target_os = "macos") {
176        //     if let Some(crtime) = crtime {
177        //         meta.crtime = crtime;
178        //     }
179        // }
180
181        if let Some(uid) = uid {
182            meta.uid = uid;
183        }
184        if let Some(gid) = gid {
185            meta.gid = gid;
186        }
187        if let Some(flags) = flags {
188            meta.flags = flags;
189        }
190        self.network_interface.update_metadata(ino, meta.clone())?;
191        Ok(meta)
192    }
