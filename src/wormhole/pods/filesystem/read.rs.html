13custom_error! {
14    /// Error describing the read syscall
15    #[derive(Clone)]
16    pub ReadError
17    WhError{source: WhError} = "{source}",
18    PullError{source: PullError} = "{source}",
19    LocalReadFailed{io: Arc<std::io::Error>} = "Local read failed: {io}",
20    CantPull = "Unable to pull file",
21    NoReadPermission = "The permissions doesn't allow to read",
22    NoFileHandle = "The file doesn't have a file handle",
23}
24
25impl From<std::io::Error> for ReadError {
26    fn from(io: std::io::Error) -> Self {
27        Self::LocalReadFailed { io: Arc::new(io) }
28    }
29}

31fn check_file_handle<'a>(
32    file_handles: &'a RwLockReadGuard<FileHandleManager>,
33    file_handle_id: UUID,
34) -> Result<&'a FileHandle, ReadError> {
35    match file_handles.handles.get(&file_handle_id) {
36        Some(&FileHandle {
37            perm: AccessMode::Write,
38            direct: _,
39            no_atime: _,
40            dirty: _,
41            ino: _,
42            signature: _,
43        }) => Err(ReadError::NoReadPermission),
44        Some(&FileHandle {
45            perm: AccessMode::Execute,
46            direct: _,
47            no_atime: _,
48            dirty: _,
49            ino: _,
50            signature: _,
51        }) => Err(ReadError::NoReadPermission),
52        None => Err(ReadError::NoFileHandle),
53        Some(file_handle) => Ok(file_handle),
54    }
55}

58    /// # Panics
59    ///
60    /// This function panics if called within an asynchronous execution
61    /// context.

63    pub fn get_file_data_sync(
64        &self,
65        ino: Ino,
66        offset: usize,
67        buf: &mut [u8],
68    ) -> Result<usize, ReadError> {
69        match self.pull_file_sync(ino)? {
70            None => Ok(self.disk.read_file(
71                &ITree::read_lock(&self.network_interface.itree, "read_file")?
72                    .get_path_from_inode_id(ino)?,
73                offset,
74                buf,
75            )?),
76            Some(data) => {
77                let size = data.len().saturating_sub(offset);
78                if size > 0 {
79                    buf[..size].copy_from_slice(&data[offset..offset + size]);
80                }
81                Ok(size)
82            }
83        }
84    }

86    /// Pull the file from the network
87    /// Affects the itree and the disk
88    ///
89    /// # Panics
90    ///
91    /// This function panics if called within an asynchronous execution
92    /// context.
93    ///
94    pub fn pull_file_sync(&self, ino: Ino) -> Result<Option<Vec<u8>>, PullError> {
95        let data = match self.network_interface.pull_file(ino)? {
96            Some(data) => data,
97            None => return Ok(None),
98        };
99
100        let (path, perms) = {
101            let itree = self.network_interface.itree.read();
102
103            let path = itree.get_path_from_inode_id(ino)?;
104            let inode = itree.get_inode(ino)?;
105            (path, inode.meta.perm)
106        };
107
108        let _created = self.disk.new_file(&path, perms);
109
110        self.disk
111            .write_file(&path, &data, 0)
112            .inspect_err(|e| log::error!("writing pulled file: {e}"))
113            .map_err(|e| PullError::WriteError { io: Arc::new(e) })?;
114
115        self.network_interface
116            .add_inode_hosts(ino, &[self.network_interface.id])
117            .expect("can't update inode hosts");
118        Ok(Some(data))
119    }

121    /// Get or pull the file from storage or network
122    ///
123    /// # Panics
124    ///
125    /// This function panics if called within an asynchronous execution
126    /// context.
127    ///
128    pub fn get_whole_file_sync(&self, ino: Ino) -> Result<File, ReadError> {
129        match self.pull_file_sync(ino)? {
130            None => self
131                .get_local_file(ino)
132                .map(|o| o.expect("promised by pull_file_sync")),
133            Some(data) => Ok(File(Arc::new(data))),
134        }
135    }

137    /// Get locally stored file as-is if it exists without accessing the network
138    /// returns Ok(Some(file)) if the file is tracked
139    /// returns Ok(None) if the file is not tracked
140    pub fn get_local_file(&self, ino: Ino) -> Result<Option<File>, ReadError> {
141        let mut buf = Vec::new();
142        let itree = self.network_interface.itree.read();
143        let size = itree.get_inode(ino).and_then(|inode| match &inode.entry {
144            FsEntry::File(hosts) => Ok(hosts
145                .contains(&self.network_interface.id)
146                .then_some(inode.meta.size as usize)),
147            _ => Err(WhError::InodeIsADirectory),
148        })?;
149        drop(itree);
150        if let Some(mut size) = size {
151            buf.resize(size, 0);
152            size = self.disk.read_file(
153                &ITree::read_lock(&self.network_interface.itree, "read_file")?
154                    .get_path_from_inode_id(ino)?,
155                0,
156                &mut buf[..],
157            )?;
158            buf.resize(size, 0);
159            Ok(Some(File(Arc::new(buf))))
160        } else {
161            Ok(None)
162        }
163    }

165    pub fn read_file(
166        &self,
167        file: Ino,
168        offset: usize,
169        buf: &mut [u8],
170        file_handle: UUID,
171    ) -> Result<usize, ReadError> {
172        {
173            let file_handles = FileHandleManager::read_lock(&self.file_handles, "read")?;
174            let _file_handle = check_file_handle(&file_handles, file_handle)?;
175        }
176
177        self.get_file_data_sync(file, offset, buf)
178    }
