32#[derive(PartialEq, Debug, Serialize, Deserialize, Clone)]
33pub enum SimpleFileType {
34    File,
35    Directory,
36    Symlink,
37}
38
39impl From<&FsEntry> for SimpleFileType {
40    fn from(entry: &FsEntry) -> SimpleFileType {
41        match entry {
42            FsEntry::File(_) => SimpleFileType::File,
43            FsEntry::Directory(_) => SimpleFileType::Directory,
44            FsEntry::Symlink(_) => SimpleFileType::Symlink,
45        }
46    }
47}
48
49impl TryFrom<std::fs::FileType> for SimpleFileType {
50    type Error = std::io::Error;
51    fn try_from(entry: std::fs::FileType) -> Result<Self, Self::Error> {
52        match (entry.is_file(), entry.is_dir(), entry.is_symlink()) {
53            (true, false, false) => Ok(SimpleFileType::File),
54            (false, true, false) => Ok(SimpleFileType::Directory),
55            (false, false, true) => Ok(SimpleFileType::Symlink),
56            _ => Err(io::ErrorKind::PermissionDenied.into()),
57        }
58    }
59}

83    pub fn get_entry_from_name(&self, parent: Ino, name: &str) -> WhResult<Option<Inode>> {
84        let itree = ITree::read_lock(&self.itree, "fs_interface.get_entry_from_name")?;
85        let p_inode = itree.get_inode(parent)?;
86        if !has_execute_perm(p_inode.meta.perm) {
87            return Ok(None);
88        }
89        Ok(Some(itree.get_inode_child_by_name(p_inode, name)?.clone()))
90    }

101    pub fn recept_inode(&self, inode: Inode) -> Result<(), MakeInodeError> {
102        self.network_interface.acknowledge_new_file(inode.clone())?;
103
104        let new_path = {
105            let itree = ITree::read_lock(&self.itree, "recept_inode")?;
106            itree.get_path_from_inode_id(inode.id)?
107        };
108
109        match &inode.entry {
110            // REVIEW - is it still useful to create an empty file in this case ?
111            FsEntry::File(hosts) if hosts.contains(&self.network_interface.hostname) => self
112                .disk
113                .new_file(&new_path, inode.meta.perm)
114                .map(|_| ())
115                .map_err(|io| MakeInodeError::LocalCreationFailed { io }),
116            FsEntry::File(_) => Ok(()),
117            FsEntry::Directory(_) => self
118                .disk
119                .new_dir(&new_path, inode.meta.perm)
120                .map_err(|io| MakeInodeError::LocalCreationFailed { io }),
121            FsEntry::Symlink(symlink) => self
122                .disk
123                .new_symlink(&new_path, inode.meta.perm, symlink)
124                .map_err(|io| MakeInodeError::LocalCreationFailed { io }),
125            // TODO - remove when merge is handled because new file should create folder
126            // FsEntry::Directory(_) => {}
127        }
128    }

130    pub fn recept_redundancy(&self, id: Ino, binary: Arc<Vec<u8>>) -> WhResult<()> {
131        let itree = ITree::write_lock(&self.itree, "recept_binary")
132            .expect("recept_binary: can't read lock itree");
133        let (path, perms) = itree
134            .get_path_from_inode_id(id)
135            .and_then(|path| itree.get_inode(id).map(|inode| (path, inode.meta.perm)))?;
136        drop(itree);
137
138        let _created = self.disk.new_file(&path, perms);
139        self.disk
140            .write_file(&path, &binary, 0)
141            .inspect_err(|e| log::error!("{e}"))
142            .expect("disk error");
143        // TODO -> in case of failure, other hosts still think this one is valid. Should send error report to the redundancy manager
144
145        ITree::write_lock(&self.itree, "recept_redundancy")?
146            .add_inode_hosts(id, vec![self.network_interface.hostname.clone()])
147            .inspect_err(|e| {
148                log::error!("Can't update (local) hosts for redundancy pulled file ({id}): {e}")
149            })
150    }

152    pub fn recept_binary(&self, id: Ino, binary: Vec<u8>) -> io::Result<()> {
153        let itree = ITree::read_lock(&self.itree, "recept_binary")
154            .expect("recept_binary: can't read lock itree");
155        let (path, perms) = match itree
156            .get_path_from_inode_id(id)
157            .and_then(|path| itree.get_inode(id).map(|inode| (path, inode.meta.perm)))
158        {
159            Ok(value) => value,
160            Err(e) => {
161                return self
162                    .network_interface
163                    .callbacks
164                    .resolve(Request::Pull(id), Err(e.into()))
165            }
166        };
167        drop(itree);
168        let _created = self.disk.new_file(&path, perms);
169        let status = self
170            .disk
171            .write_file(&path, &binary, 0)
172            .inspect_err(|e| log::error!("writing pulled file: {e}"))
173            .map_err(|e| PullError::WriteError { io: Arc::new(e) });
174        let _ = self.network_interface.callbacks.resolve(
175            Request::Pull(id),
176            status
177                .as_ref()
178                .map_err(|e| e.clone())
179                .map(|_| Some(Arc::new(binary))),
180        );
181        status.map_err(io::Error::other)?;
182        self.network_interface
183            .add_inode_hosts(id, vec![self.network_interface.hostname.clone()])
184            .expect("can't update inode hosts");
185        Ok(())
186    }

188    pub fn recept_edit_hosts(&self, id: Ino, hosts: Vec<Address>) -> WhResult<()> {
189        if !hosts.contains(&self.network_interface.hostname) {
190            let path =
191                ITree::read_lock(&self.itree, "recept_edit_hosts")?.get_path_from_inode_id(id)?;
192            if let Err(e) = self.disk.remove_file(&path) {
193                log::debug!("recept_edit_hosts: can't delete file. {}", e);
194            }
195        }
196        self.network_interface.acknowledge_hosts_edition(id, hosts)
197    }

199    pub fn recept_revoke_hosts(
200        &self,
201        id: Ino,
202        host: String,
203        meta: Metadata,
204    ) -> Result<(), AcknoledgeSetAttrError> {
205        let needs_delete = host != self.network_interface.hostname;
206        self.acknowledge_metadata(id, meta)?;
207        self.network_interface
208            .acknowledge_hosts_edition(id, vec![host])
209            .map_err(|source| AcknoledgeSetAttrError::WhError { source })?;
210        if needs_delete {
211            // TODO: recept_revoke_hosts, for the redudancy, should recieve the written text (data from write) instead of deleting and adding it back completely with apply_redudancy
212            if let Err(e) = self.disk.remove_file(
213                &ITree::read_lock(&self.itree, "recept_revoke_hosts")?
214                    .get_path_from_inode_id(id)?,
215            ) {
216                log::debug!("recept_revoke_hosts: can't delete file. {}", e);
217            }
218        }
219        Ok(())
220    }

242    pub fn send_filesystem(&self, to: Address) -> WhResult<()> {
243        let itree = ITree::read_lock(&self.itree, "fs_interface::send_filesystem")?;
244        let global_config_file_size = itree
245            .get_inode(GLOBAL_CONFIG_INO)
246            .map(|inode| inode.meta.size)
247            .ok();
248        let global_config_path = if global_config_file_size.is_some() {
249            Some(WhPath::try_from(GLOBAL_CONFIG_FNAME).expect("GLOBAL_CONFIG_FNAME const error"))
250        } else {
251            None
252        };
253        drop(itree);
254        log::info!("reading global config at {global_config_path:?}");
255
256        let mut global_config_bytes = Vec::new();
257        if let Some(global_config_file_size) = global_config_file_size {
258            global_config_bytes.resize(global_config_file_size as usize, 0);
259
260            if let Some(global_config_path) = global_config_path {
261                self.disk
262                    .read_file(&global_config_path, 0, &mut global_config_bytes)
263                    .expect("disk can't read file (global condfig)");
264            }
265        }
266        self.network_interface.send_itree(to, global_config_bytes)
267    }

269    pub fn send_file(&self, inode: Ino, to: Address) -> io::Result<()> {
270        let itree = ITree::read_lock(&self.itree, "send_itree").map_err(io::Error::other)?;
271        let path = itree
272            .get_path_from_inode_id(inode)
273            .map_err(io::Error::other)?;
274        let mut size = itree.get_inode(inode).map_err(io::Error::other)?.meta.size as usize;
275        let mut data = vec![0; size];
276        size = self.disk.read_file(&path, 0, &mut data)?;
277        data.resize(size, 0);
278        self.network_interface
279            .send_file(inode, data, to)
280            .map_err(io::Error::other)
281    }

283    pub fn read_local_file(&self, inode: Ino) -> WhResult<Vec<u8>> {
284        let itree = ITree::read_lock(&self.itree, "send_itree")?;
285        let path = itree
286            .get_path_from_inode_id(inode)
287            .map_err(|_| crate::error::WhError::InodeNotFound)?;
288        let size = itree.get_inode(inode)?.meta.size;
289        drop(itree);
290
291        let mut buff = vec![0; size as usize];
292        self.disk
293            .read_file(&path, 0, &mut buff)
294            .map_err(|_| crate::error::WhError::InodeNotFound)?;
295        Ok(buff)
296    }

298    //REVIEW - I don't really like to lock the arbo here, but it's the only way to get the inode countwithout just using an arbitrary high number
299    /// Get complete filesystem size information including inode counts
300    pub fn get_size_info(&self) -> io::Result<crate::pods::disk_managers::DiskSizeInfo> {
301        let mut disk_info = self.disk.size_info()?;
302
303        let itree = ITree::read_lock(&self.itree, "fs_interface::get_size_info").map_err(|_| {
304            io::Error::other(crate::error::WhError::WouldBlock {
305                called_from: "fs_interface::get_size_info".to_string(),
306            })
307        })?;
308        let files = itree.iter().count() as u64;
309        let next_ino = itree.next_ino.start;
310
311        let ffree = if next_ino < u64::MAX / 2 {
312            u64::MAX - next_ino
313        } else {
314            1_000_000_000
315        };
316
317        disk_info.files = files;
318        disk_info.ffree = ffree;
319
320        Ok(disk_info)
321    }
