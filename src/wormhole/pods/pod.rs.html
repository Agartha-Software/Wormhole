77async fn initiate_connection(
78    mountpoint: &WhPath,
79    local_config: &LocalConfig,
80    global_config: &GlobalConfig,
81    receiver_in: &UnboundedSender<FromNetworkMessage>,
82    receiver_out: UnboundedReceiver<FromNetworkMessage>,
83) -> Result<PodPrototype, UnboundedReceiver<FromNetworkMessage>> {
84    if global_config.general.entrypoints.len() >= 1 {
85        for first_contact in &global_config.general.entrypoints {
86            match PeerIPC::connect(first_contact.to_owned(), local_config, receiver_in.clone())
87                .await
88            {
89                Err(HandshakeError::CouldntConnect) => continue,
90                Err(e) => {
91                    log::error!("{first_contact}: {e}");
92                    return Err(receiver_out);
93                }
94                Ok((ipc, accept)) => {
95                    return if let Some(urls) =
96                        accept.urls.into_iter().skip(1).fold(Some(vec![]), |a, b| {
97                            a.and_then(|mut a| {
98                                a.push(b?);
99                                Some(a)
100                            })
101                        }) {
102                        let mut local_config = local_config.clone();
103                        if let Some(rename) = accept.rename {
104                            local_config.general.hostname = rename;
105                        }
106
107                        match PeerIPC::peer_startup(
108                            urls,
109                            local_config.general.hostname.clone(),
110                            accept.hostname,
111                            receiver_in.clone(),
112                        )
113                        .await
114                        {
115                            Ok(mut other_ipc) => {
116                                other_ipc.insert(0, ipc);
117                                Ok(PodPrototype {
118                                    arbo: accept.arbo,
119                                    peers: other_ipc,
120                                    global_config: accept.config,
121                                    local_config,
122                                    mountpoint: mountpoint.clone(),
123                                    receiver_out,
124                                    receiver_in: receiver_in.clone(),
125                                })
126                            }
127                            Err(e) => {
128                                log::error!("a peer failed: {e}");
129                                Err(receiver_out)
130                            }
131                        }
132                    } else {
133                        log::error!("Peers do not all have a url!");
134                        Err(receiver_out)
135                    };
136                }
137            }
138        }
139        info!("None of the known address answered correctly, starting a FS.")
140    }
141    Err(receiver_out)
142}

161/// Create all the directories present in Arbo. (not the files)
162///
163/// Required at setup to resolve issue #179
164/// (files pulling need the parent folder to be already present)
165fn create_all_dirs(arbo: &Arbo, from: InodeId, disk: &dyn DiskManager) -> io::Result<()> {
166    let from = arbo.n_get_inode(from).map_err(|e| e.into_io())?;
167
168    return match &from.entry {
169        FsEntry::File(_) => Ok(()),
170        FsEntry::Directory(children) => {
171            let current_path = arbo
172                .n_get_path_from_inode_id(from.id)
173                .map_err(|e| e.into_io())?;
174            disk.new_dir(&current_path, from.meta.perm).or_else(|e| {
175                if e.kind() == io::ErrorKind::AlreadyExists {
176                    Ok(())
177                } else {
178                    Err(e)
179                }
180            })?;
181
182            for child in children {
183                create_all_dirs(arbo, *child, disk)?
184            }
185            Ok(())
186        }
187    };
188}

191    pub async fn new(
192        global_config: GlobalConfig,
193        local_config: LocalConfig,
194        mountpoint: WhPath,
195        server: Arc<Server>,
196    ) -> io::Result<Self> {
197        let global_config = global_config;
198
199        log::trace!("mount point {}", mountpoint);
200        let (receiver_in, receiver_out) = mpsc::unbounded_channel();
201
202        let proto = match initiate_connection(
203            &mountpoint,
204            &local_config,
205            &global_config,
206            &receiver_in,
207            receiver_out,
208        )
209        .await
210        {
211            Ok(proto) => proto,
212            Err(receiver_out) => {
213                if global_config.general.entrypoints.len() > 0 {
214                    // NOTE - temporary fix
215                    // made to help with tests and debug
216                    // choice not to fail should later be supported by the cli
217                    log::error!("No peers answered. Stopping.");
218                    return Err(io::Error::new(
219                        io::ErrorKind::Other,
220                        "None of the specified peers could answer",
221                    ));
222                }
223                let arbo = generate_arbo(&mountpoint, &local_config.general.hostname)
224                    .unwrap_or(Arbo::new());
225                PodPrototype {
226                    arbo,
227                    peers: vec![],
228                    global_config,
229                    local_config,
230                    mountpoint,
231                    receiver_out,
232                    receiver_in,
233                }
234            }
235        };
236
237        Self::realize(proto, server).await
238    }

240    async fn realize(proto: PodPrototype, server: Arc<Server>) -> io::Result<Self> {
241        let (senders_in, senders_out) = mpsc::unbounded_channel();
242
243        let (redundancy_tx, redundancy_rx) = mpsc::unbounded_channel();
244
245        #[cfg(target_os = "linux")]
246        let disk_manager = Box::new(UnixDiskManager::new(&proto.mountpoint)?);
247        #[cfg(target_os = "windows")]
248        let disk_manager = Box::new(DummyDiskManager::new(&proto.mountpoint)?);
249
250        create_all_dirs(&proto.arbo, ROOT, disk_manager.as_ref())
251            .inspect_err(|e| log::error!("unable to create_all_dirs: {e}"))
252            .map_err(|e| std::io::Error::new(e.kind(), format!("create_all_dirs: {e}")))?;
253
254        if let Ok(perms) = proto
255            .arbo
256            .get_inode(GLOBAL_CONFIG_INO)
257            .map(|inode| inode.meta.perm)
258        {
259            let _ = disk_manager.new_file(&GLOBAL_CONFIG_FNAME.into(), perms);
260            disk_manager
261                .write_file(
262                    &GLOBAL_CONFIG_FNAME.into(),
263                    toml::to_string(&proto.global_config)
264                        .expect("infallible")
265                        .as_bytes(),
266                    0,
267                )
268                .map_err(|e| {
269                    std::io::Error::new(e.kind(), format!("write_file(global_config): {e}"))
270                })?;
271        }
272
273        if let Ok(perms) = proto
274            .arbo
275            .get_inode(LOCAL_CONFIG_INO)
276            .map(|inode| inode.meta.perm)
277        {
278            let _ = disk_manager.new_file(&LOCAL_CONFIG_FNAME.into(), perms);
279            disk_manager
280                .write_file(
281                    &LOCAL_CONFIG_FNAME.into(),
282                    toml::to_string(&proto.local_config)
283                        .expect("infallible")
284                        .as_bytes(),
285                    0,
286                )
287                .map_err(|e| {
288                    std::io::Error::new(e.kind(), format!("write_file(local_config): {e}"))
289                })?;
290        }
291
292        let url = proto.local_config.general.url.clone();
293
294        let arbo: Arc<RwLock<Arbo>> = Arc::new(RwLock::new(proto.arbo));
295        let local = Arc::new(RwLock::new(proto.local_config));
296        let global = Arc::new(RwLock::new(proto.global_config));
297
298        let network_interface = Arc::new(NetworkInterface::new(
299            arbo.clone(),
300            url,
301            senders_in.clone(),
302            redundancy_tx.clone(),
303            Arc::new(RwLock::new(proto.peers)),
304            local.clone(),
305            global.clone(),
306        ));
307
308        let fs_interface = Arc::new(FsInterface::new(
309            network_interface.clone(),
310            disk_manager,
311            arbo.clone(),
312        ));
313
314        // Start ability to recieve messages
315        let network_airport_handle = tokio::spawn(NetworkInterface::network_airport(
316            proto.receiver_out,
317            fs_interface.clone(),
318        ));
319
320        // Start ability to send messages
321        let peer_broadcast_handle = tokio::spawn(NetworkInterface::contact_peers(
322            network_interface.peers.clone(),
323            senders_out,
324        ));
325
326        let new_peer_handle = tokio::spawn(NetworkInterface::incoming_connections_watchdog(
327            server,
328            proto.receiver_in.clone(),
329            network_interface.clone(),
330        ));
331
332        let peers = network_interface.peers.clone();
333
334        let redundancy_worker_handle = tokio::spawn(redundancy_worker(
335            redundancy_rx,
336            network_interface.clone(),
337            fs_interface.clone(),
338        ));
339
340        Ok(Self {
341            network_interface,
342            fs_interface: fs_interface.clone(),
343            mountpoint: proto.mountpoint.clone(),
344            peers,
345            #[cfg(target_os = "linux")]
346            fuse_handle: mount_fuse(&proto.mountpoint, fs_interface.clone())
347                .map_err(|e| std::io::Error::new(e.kind(), format!("mount_fuse: {e}")))?,
348            #[cfg(target_os = "windows")]
349            fsp_host: mount_fsp(&proto.mountpoint, fs_interface.clone())
350                .map_err(|e| std::io::Error::new(e.kind(), format!("mount_fsp: {e}")))?,
351            network_airport_handle,
352            peer_broadcast_handle,
353            new_peer_handle,
354            local_config: local.clone(),
355            global_config: global.clone(),
356            redundancy_worker_handle,
357        })
358    }

362    pub fn get_file_hosts(&self, path: WhPath) -> Result<Vec<Address>, PodInfoError> {
363        let path_string = path.to_string();
364        log::info!("Get file host at: {}", path_string);
365        let path = path_string
366            .strip_prefix(&self.mountpoint.to_string())
367            .ok_or(PodInfoError::WrongFileType {
368                detail: "Asked path is a directory (directories have no hosts)".to_owned(),
369            })?;
370        log::info!("Get file host at local path: {}", path);
371
372        let binding = Arbo::n_read_lock(&self.network_interface.arbo, "Pod::get_info")?;
373        let entry = &binding
374            .get_inode_from_path(&WhPath::from(path))
375            .map_err(|_| PodInfoError::FileNotFound)?
376            .entry;
377
378        match entry {
379            FsEntry::File(hosts) => Ok(hosts.clone()),
380            FsEntry::Directory(_) => Err(PodInfoError::WrongFileType {
381                detail: "Asked path is a directory (directories have no hosts)".to_owned(),
382            }),
383        }
384    }

387        &self,

404    /// given ino is not checked -> must exist in arbo
405    fn recurse_tree(arbo: &Arbo, ino: InodeId, indentation: u8) -> Vec<TreeLine> {
406        let entry = &arbo
407            .n_get_inode(ino)
408            .expect("recurse_tree: ino not found")
409            .entry;
410        let path = arbo
411            .n_get_path_from_inode_id(ino)
412            .expect("recurse_tree: unable to get path");
413        match entry {
414            FsEntry::File(hosts) => vec![(indentation, ino, path, hosts.clone())],
415            FsEntry::Directory(children) => children
416                .iter()
417                .map(|c| Pod::recurse_tree(arbo, *c, indentation + 1))
418                .flatten()
419                .collect::<Vec<TreeLine>>(),
420        }
421    }

425    /// for a given file, will try to send it to one host, trying each until succes
426    async fn send_file_to_possible_hosts(
427        &self,
428        possible_hosts: &Vec<Address>,
429        ino: InodeId,
430    ) -> Result<(), PodStopError> {
431        let file_content =
432            self.fs_interface
433                .read_local_file(ino)
434                .map_err(|e| PodStopError::FileNotReadable {
435                    file: ino,
436                    reason: e.to_string(),
437                })?;
438        let file_content = Arc::new(file_content);
439
440        for host in possible_hosts {
441            let (status_tx, mut status_rx) = tokio::sync::mpsc::unbounded_channel::<WhResult<()>>();
442
443            self.network_interface
444                .to_network_message_tx
445                .send(ToNetworkMessage::SpecificMessage(
446                    (
447                        // NOTE - file_content clone is not efficient, but no way to avoid it for now
448                        MessageContent::RedundancyFile(ino, file_content.clone()),
449                        Some(status_tx.clone()),
450                    ),
451                    vec![host.clone()],
452                ))
453                .expect("to_network_message_tx closed.");
454
455            if let Some(Ok(())) = status_rx.recv().await {
456                self.network_interface
457                    .to_network_message_tx
458                    .send(ToNetworkMessage::BroadcastMessage(
459                        MessageContent::EditHosts(ino, vec![host.clone()]),
460                    ))
461                    .expect("to_network_message_tx closed.");
462                return Ok(());
463            }
464        }
465        Err(PodStopError::FileNotSent { file: ino })
466    }

468    /// Gets every file hosted by this pod only and sends them to other pods
469    async fn send_files_when_stopping(&self, arbo: &Arbo, peers: Vec<Address>) {
470        futures_util::future::join_all(
471            arbo.files_hosted_only_by(
472                &self
473                    .network_interface
474                    .local_config
475                    .read()
476                    .general
477                    .hostname
478                    .clone(),
479            )
480            .filter_map(|inode| {
481                if inode.id == GLOBAL_CONFIG_INO
482                    || inode.id == LOCAL_CONFIG_INO
483                    || inode.id == ARBO_FILE_INO
484                {
485                    None
486                } else {
487                    Some(inode.id)
488                }
489            })
490            .map(|id| self.send_file_to_possible_hosts(&peers, id)),
491        )
492        .await
493        .iter()
494        .for_each(|e| {
495            if let Err(e) = e {
496                log::warn!("{e:?}")
497            }
498        });
499    }

501    pub async fn stop(self) -> Result<(), PodStopError> {
502        // TODO
503        // in actual state, all operations (request from network other than just pulling the asked files)
504        // made after calling this function but before dropping the pod are undefined behavior.
505
506        // drop(self.fuse_handle); // FIXME - do something like block the filesystem
507
508        let arbo = Arbo::n_read_lock(&self.network_interface.arbo, "Pod::Pod::stop(1)")?;
509
510        let peers: Vec<Address> = self
511            .peers
512            .read()
513            .iter()
514            .map(|peer| peer.hostname.clone())
515            .collect();
516
517        self.send_files_when_stopping(&arbo, peers).await;
518        let arbo_bin = bincode::serialize(&*arbo).expect("can't serialize arbo to bincode");
519        drop(arbo);
520
521        self.network_interface
522            .to_network_message_tx
523            .send(ToNetworkMessage::BroadcastMessage(
524                MessageContent::Disconnect(
525                    self.network_interface
526                        .local_config
527                        .read()
528                        .general
529                        .hostname
530                        .clone(),
531                ),
532            ))
533            .expect("to_network_message_tx closed.");
534
535        let Self {
536            network_interface: _,
537            fs_interface,
538            mountpoint: _,
539            peers,
540            #[cfg(target_os = "linux")]
541            fuse_handle,
542            #[cfg(target_os = "windows")]
543            fsp_host,
544            network_airport_handle,
545            peer_broadcast_handle,
546            new_peer_handle,
547            redundancy_worker_handle: _,
548            global_config: _,
549            local_config: _,
550        } = self;
551
552        #[cfg(target_os = "linux")]
553        drop(fuse_handle);
554        #[cfg(target_os = "windows")]
555        drop(fsp_host);
556
557        fs_interface
558            .disk
559            .write_file(&ARBO_FILE_FNAME.into(), &arbo_bin, 0)
560            .map_err(|io| PodStopError::ArboSavingFailed { source: io })?;
561
562        *peers.write() = Vec::new(); // dropping PeerIPCs
563        network_airport_handle.abort();
564        new_peer_handle.abort();
565        peer_broadcast_handle.abort();
566        Ok(())
567    }

581            .peers

593            hostname: self
594                .network_interface
595                .hostname()
596                .expect("Can't lock network"),
597            name: "".to_string(), //TODO to delete
