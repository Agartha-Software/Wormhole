15pub async fn new(args: PodArgs) -> CliResult<Pod> {
16    let (global_config, local_config, server, mount_point) = pod_value(&args).await?;
17    Pod::new(
18        // local_config.general.name.clone(),
19        global_config,
20        local_config.clone(),
21        mount_point,
22        server.clone(),
23    )
24    .await
25    .map_err(|e| CliError::PodCreationFailed { reason: e })
26    .inspect_err(|e| log::error!("Pod creation failed: {}", e))
27}

29fn add_hosts(
30    mut global_config: GlobalConfig,
31    url: String,
32    mut additional_hosts: Vec<String>,
33) -> GlobalConfig {
34    if url.is_empty() && additional_hosts.is_empty() {
35        return global_config;
36    }
37
38    additional_hosts.insert(0, url);
39    global_config.general.entrypoints.extend(additional_hosts);
40    global_config
41}

43async fn pod_value(args: &PodArgs) -> CliResult<(GlobalConfig, LocalConfig, Arc<Server>, WhPath)> {
44    log::info!("args: {args:?}");
45    let path = args
46        .mountpoint
47        .as_ref()
48        .and_then(|path| {
49            let (parent, folder) = path.split_folder_file();
50            std::fs::canonicalize(&parent)
51                .ok()
52                .map(|p| PathBuf::from(p).join(folder))
53        })
54        .ok_or(CliError::InvalidArgument {
55            arg: "path".to_owned(),
56        })?;
57
58    log::info!("canonical: {:?}", path);
59
60    let address = "0.0.0.0:".to_owned() + &args.port;
61    let local_cfg_path = path.join(LOCAL_CONFIG_FNAME);
62    let global_cfg_path = path.join(GLOBAL_CONFIG_FNAME);
63
64    let mut local_config: LocalConfig = LocalConfig::read(&local_cfg_path).unwrap_or_default();
65    local_config.general.hostname = args
66        .hostname
67        .clone()
68        .or_else(|| gethostname().into_string().ok())
69        .ok_or(CliError::Message {
70            reason: "no valid hostname".to_owned(),
71        })?;
72    local_config.general.url = Some(
73        args.listen_url
74            .as_ref()
75            .unwrap_or(&local_config.general.hostname)
76            .clone(),
77    );
78    let server: Arc<Server> = Arc::new(Server::setup(&address).await?);
79
80    let global_config: GlobalConfig = GlobalConfig::read(global_cfg_path).unwrap_or_default();
81    let global_config = add_hosts(
82        global_config,
83        args.url.clone().unwrap_or("".to_string()),
84        args.additional_hosts.clone(),
85    );
86
87    Ok((global_config, local_config, server, path.as_os_str().into()))
88}
