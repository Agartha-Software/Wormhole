5use crate::commands;
6use crate::config::types::Config;
7use std::error::Error;
8use std::fs;

10#[derive(PartialEq)]
11pub enum Mode {
12    /// Simply remove the pod from the network without losing any data from the network
13    /// and leaving behind any data that was stored on the pod
14    Simple,
15    /// remove the pod from the network without losing any data on the network,
16    /// and clone all data from the network into the folder where the pod was
17    /// making this folder into a real folder
18    Clone,
19    /// remove the pod from the network and delete any data that was stored in the pod
20    Clean,
21    /// remove this pod from the network without distributing its data to other nodes
22    Take,
23}
24
25#[must_use]
26pub fn remove(path: &std::path::PathBuf, mode: Mode) -> Result<(), Box<dyn Error>> {
27    if mode != Mode::Take {
28        println!("todo!: implement redistribute");
29    }
30    if mode == Mode::Clone {
31        todo!("clone")
32    }
33
34    let name = crate::config::Network::read(path.join(".wormhole/network.toml"))?.name;
35
36    commands::unregister(&name)?;
37    fs::remove_dir_all((&path).join(".wormhole"))?;
38    if mode == Mode::Clean {
39        fs::remove_dir_all(path)?;
40    }
41    Ok(())
42}
