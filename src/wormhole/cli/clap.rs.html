56fn canonicalize(path: PathBuf) -> std::io::Result<PathBuf> {
57    std::fs::canonicalize(path)
58}
59
60fn parse_canonicalize(path_str: &str) -> std::io::Result<PathBuf> {
61    canonicalize(PathBuf::from(path_str))
62}
63
64// like canonicalize but doesn't check if the final element exist
65pub fn parse_canonicalize_non_existant(path_str: &str) -> std::io::Result<PathBuf> {
66    let path = PathBuf::from(path_str);
67
68    if path.exists() {
69        return canonicalize(path);
70    }
71
72    let name = path.file_name().ok_or(std::io::Error::new(
73        std::io::ErrorKind::NotFound,
74        "Path doesn't exist",
75    ))?;
76
77    let mut parent = match path.parent() {
78        Some(parent) => canonicalize(parent.to_path_buf())?,
79        None => std::env::current_dir()?,
80    };
81    parent.push(name);
82    Ok(parent)
83}
