130pub enum Mode {
131    /// Simply remove the pod from the network without losing any data from the network
132    /// and leaving behind any data that was stored on the pod
133    Simple,
134    /// Remove the pod from the network without losing any data on the network,
135    /// and clone all data from the network into the folder where the pod was
136    /// making this folder into a real folder
137    Clone,
138    /// Remove the pod from the network and delete any data that was stored in the pod
139    Clean,
140    /// Remove this pod from the network without distributing its data to other nodes
141    Take,
142}
