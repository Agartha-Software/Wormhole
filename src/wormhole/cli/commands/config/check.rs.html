1use interprocess::local_socket::tokio::Stream;
2use std::io;
3
4use crate::{
5    cli::{
6        connection::{recieve_answer, send_command},
7        ConfigType, IdentifyPodAndConfigArgs,
8    },
9    ipc::{
10        answers::{CheckConfigAnswer, ConfigFileError},
11        commands::{Command, PodId},
12    },
13};
14
15pub async fn check(args: IdentifyPodAndConfigArgs, mut stream: Stream) -> io::Result<String> {
16    let pod = PodId::from(args.group);
17
18    send_command(
19        Command::CheckConfig(pod, args.config_type.clone()),
20        &mut stream,
21    )
22    .await?;
23
24    match recieve_answer::<CheckConfigAnswer>(&mut stream).await? {
25        CheckConfigAnswer::Success if args.config_type == ConfigType::Local => Ok("The pod local configuration file is valid!".into()),
26        CheckConfigAnswer::Success if args.config_type == ConfigType::Global => Ok("The pod global configuration file is valid!".into()),
27        CheckConfigAnswer::Success => Ok("The pod configuration files are valid!".into()),
28        CheckConfigAnswer::PodNotFound => Err(io::Error::new(
29            io::ErrorKind::NotFound,
30            "The given pod couldn't be found.",
31        )),
32        CheckConfigAnswer::ConfigFileError(ConfigFileError::MissingGlobal) => Err(io::Error::new(
33            io::ErrorKind::NotFound,
34            "The given pod doesn't have a `global_config.toml` file to check.",
35        )),
36        CheckConfigAnswer::ConfigFileError(ConfigFileError::MissingLocal) => Err(io::Error::new(
37            io::ErrorKind::NotFound,
38            "The given pod doesn't have a `local_config.toml` file to check.",
39        )),
40        CheckConfigAnswer::ConfigFileError(ConfigFileError::MissingBoth) => Err(io::Error::new(
41            io::ErrorKind::NotFound,
42            "The given pod doesn't have a either a `local_config.toml` or a `global_config.toml` file to check.",
43        )),
44        CheckConfigAnswer::ConfigFileError(ConfigFileError::InvalidGlobal(error)) => Err(io::Error::new(
45            io::ErrorKind::InvalidData,
46            format!("Global configuration validation failed:\n{error}"),
47        )),
48        CheckConfigAnswer::ConfigFileError(ConfigFileError::InvalidLocal(error)) => Err(io::Error::new(
49            io::ErrorKind::InvalidData,
50            format!("Local configuration validation failed:\n{error}"),
51        )),
52        CheckConfigAnswer::ConfigFileError(ConfigFileError::InvalidBoth(local_error, global_error)) => Err(io::Error::new(
53            io::ErrorKind::InvalidData,
54            format!("Local configuration validation failed:\n{local_error}\n\nGlobal configuration validation failed:\n{global_error}"),
55        )),
56    }
57}
