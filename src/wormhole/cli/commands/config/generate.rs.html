1use std::io;
2
3use interprocess::local_socket::tokio::Stream;
4
5use crate::{
6    cli::connection::{recieve_answer, send_command},
7    cli::{ConfigType, GenerateConfigArgs},
8    ipc::{
9        answers::GenerateConfigAnswer,
10        commands::{Command, PodId},
11    },
12};
13
14pub async fn generate(args: GenerateConfigArgs, mut stream: Stream) -> io::Result<String> {
15    let pod = PodId::from(args.group);
16
17    send_command(
18        Command::GenerateConfig(pod, args.force, args.config_type),
19        &mut stream,
20    )
21    .await?;
22
23    match recieve_answer::<GenerateConfigAnswer>(&mut stream).await? {
24        GenerateConfigAnswer::Success => Ok("Pod's configuration created successfully!".into()),
25        GenerateConfigAnswer::SuccessDefault => Ok("Default configuration created successfully!".into()),
26        GenerateConfigAnswer::PodNotFound => Err(io::Error::new(
27            io::ErrorKind::NotFound,
28            "The given pod couldn't be found.",
29        )),
30        GenerateConfigAnswer::NotADirectory => Err(io::Error::new(
31            io::ErrorKind::NotADirectory,
32            "The given directory doesn't exist.",
33        )),
34        GenerateConfigAnswer::ConfigBlock => Err(io::Error::new(
35            io::ErrorKind::WouldBlock,
36            "Failed to access the pod configuration.",
37        )),
38        GenerateConfigAnswer::WriteFailed(err, ConfigType::Local) => Err(io::Error::new(
39            io::ErrorKind::Interrupted,
40            format!("Failed to write the local configuration: {err}"),
41        )),
42        GenerateConfigAnswer::WriteFailed(err, ConfigType::Global) => Err(io::Error::new(
43            io::ErrorKind::Interrupted,
44            format!("Failed to write the global configuration: {err}"),
45        )),
46        GenerateConfigAnswer::WriteFailed(err, ConfigType::Both) => Err(io::Error::new(
47            io::ErrorKind::Interrupted,
48            format!("Failed to write the local configuration: {err}"),
49        )),
50        GenerateConfigAnswer::CantOverwrite(ConfigType::Local) => Err(io::Error::new(
51            io::ErrorKind::AlreadyExists,
52                "Couldn't write the local configuration file, it already exist...\n`--force` to overwrite existing files",
53        )),
54        GenerateConfigAnswer::CantOverwrite(ConfigType::Global) => Err(io::Error::new(
55            io::ErrorKind::AlreadyExists,
56                "Couldn't write the global configuration file, it already exist...\n`--force` to overwrite existing files",
57        )),
58        GenerateConfigAnswer::CantOverwrite(ConfigType::Both) => Err(io::Error::new(
59            io::ErrorKind::AlreadyExists,
60                "Couldn't write the configuration files, they already exist...\n`--force` to overwrite existing files",
61        )),
62    }
63}
