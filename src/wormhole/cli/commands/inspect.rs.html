12fn display_peers(peers: Vec<PeerInfo>) -> String {
13    if peers.is_empty() {
14        "[ ]".to_string()
15    } else {
16        let mut string = String::from("[");
17        for (idx, peer) in peers.iter().enumerate() {
18            string.push_str("\n      { ");
19            string.push_str(&peer.to_string());
20            string.push_str(" }");
21            if idx + 1 != peers.len() {
22                string.push(',');
23            }
24        }
25        string.push_str("\n   ]");
26        string
27    }
28}

30pub async fn inspect(args: IdentifyPodArgs, mut stream: Stream) -> Result<String, io::Error> {
31    let id = PodId::from(args);
32
33    send_command(Command::Inspect(id), &mut stream).await?;
34    match recieve_answer::<InspectAnswer>(&mut stream).await? {
35        InspectAnswer::Information(info) => Ok(format!(
36            "Pod informations: {}\n\
37            \x20  Name:\t\t{}\n\
38            \x20  Mount:\t\t{:#?}\n\
39            \x20  Hostname:\t\t{}\n\
40            \x20  Url:\t\t\t{}\n\
41            \x20  Bound Address:\t{}\n\
42            \x20  Connected peers:\t{}",
43            if info.frozen { "Frozen" } else { "Running" },
44            info.name,
45            info.mount,
46            info.hostname,
47            info.public_url.unwrap_or("[ None ]".to_string()),
48            info.bound_socket,
49            if info.frozen {
50                "Disconnected (Frozen)".to_string()
51            } else {
52                display_peers(info.connected_peers)
53            }
54        )),
55        InspectAnswer::PodNotFound => Err(io::Error::new(
56            io::ErrorKind::NotFound,
57            "The given pod couldn't be found.",
58        )),
59    }
60}
